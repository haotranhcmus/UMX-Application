# 3.3 Styling

> **Mục tiêu**: Style components với StyleSheet và hiểu responsive design trong React Native.

---

## 📖 Tổng Quan

React Native sử dụng **JavaScript objects** để style thay vì CSS. Cú pháp gần giống CSS nhưng dùng **camelCase** thay vì kebab-case.

| CSS                  | React Native          |
| -------------------- | --------------------- |
| `background-color`   | `backgroundColor`     |
| `font-size`          | `fontSize`            |
| `border-radius`      | `borderRadius`        |
| `text-align: center` | `textAlign: 'center'` |
| `px`, `rem`, `%`     | Số (dp), `%`          |

## 1. StyleSheet.create()

### 1.1 Cơ Bản

```tsx
import React from "react";
import { View, Text, StyleSheet } from "react-native";

const StyleSheetDemo = () => {
  return (
    <View style={styles.container}>
      <Text style={styles.title}>StyleSheet Demo</Text>
      <Text style={styles.body}>Body text styled with StyleSheet</Text>
    </View>
  );
};

// ✅ StyleSheet.create() — validated at creation, better performance
const styles = StyleSheet.create({
  container: {
    flex: 1,
    padding: 16,
    backgroundColor: "#fff",
  },
  title: {
    fontSize: 24,
    fontWeight: "bold",
    color: "#333",
    marginBottom: 8,
  },
  body: {
    fontSize: 16,
    lineHeight: 24,
    color: "#666",
  },
});

export default StyleSheetDemo;
```

### 1.2 Inline Styles vs StyleSheet

```tsx
import React from "react";
import { View, Text, StyleSheet } from "react-native";

const InlineVsStyleSheet = () => {
  return (
    <View style={styles.container}>
      {/* ❌ Inline style — tạo object mới mỗi render */}
      <Text style={{ fontSize: 16, color: "red", fontWeight: "bold" }}>
        Inline Style (mỗi render tạo object mới)
      </Text>

      {/* ✅ StyleSheet — object tạo 1 lần, tái sử dụng */}
      <Text style={styles.text}>StyleSheet (tạo 1 lần, reference ổn định)</Text>

      {/* ✅ OK dùng inline cho dynamic values */}
      <View
        style={[styles.box, { backgroundColor: true ? "#4CAF50" : "#f44336" }]}
      >
        <Text style={styles.boxText}>Dynamic color inline OK</Text>
      </View>
    </View>
  );
};

const styles = StyleSheet.create({
  container: { padding: 16, gap: 12 },
  text: { fontSize: 16, color: "#007AFF", fontWeight: "bold" },
  box: { padding: 12, borderRadius: 8 },
  boxText: { color: "#fff", fontWeight: "600" },
});
```

### 1.3 Combining Styles

```tsx
import React from "react";
import { View, Text, StyleSheet } from "react-native";

interface BadgeProps {
  text: string;
  type: "success" | "error" | "warning";
  large?: boolean;
}

const Badge = ({ text, type, large = false }: BadgeProps) => {
  return (
    // ✅ Mảng styles — áp dụng từ trái sang phải (sau override trước)
    <View
      style={[
        styles.badge, // Base style
        styles[type], // Type-specific style
        large && styles.large, // Conditional style
      ]}
    >
      <Text style={[styles.badgeText, type === "error" && { color: "#fff" }]}>
        {text}
      </Text>
    </View>
  );
};

const CombiningStylesDemo = () => (
  <View style={styles.container}>
    <Badge text="Success" type="success" />
    <Badge text="Error" type="error" />
    <Badge text="Warning Large" type="warning" large />
  </View>
);

const styles = StyleSheet.create({
  container: { padding: 16, flexDirection: "row", gap: 8, flexWrap: "wrap" },
  badge: {
    paddingHorizontal: 12,
    paddingVertical: 6,
    borderRadius: 16,
    borderWidth: 1,
  },
  // Type styles
  success: { backgroundColor: "#d4edda", borderColor: "#28a745" },
  error: { backgroundColor: "#dc3545", borderColor: "#dc3545" },
  warning: { backgroundColor: "#fff3cd", borderColor: "#ffc107" },
  // Size
  large: { paddingHorizontal: 20, paddingVertical: 10 },
  badgeText: { fontSize: 13, fontWeight: "600", color: "#333" },
});
```

### 1.4 Conditional & Platform-specific Styles

```tsx
import React from "react";
import { View, Text, StyleSheet, Platform } from "react-native";

interface CardProps {
  isActive: boolean;
  disabled?: boolean;
}

const PlatformCard = ({ isActive, disabled = false }: CardProps) => {
  return (
    <View
      style={[
        styles.card,
        isActive && styles.activeCard,
        disabled && styles.disabledCard,
      ]}
    >
      <Text style={[styles.text, disabled && styles.disabledText]}>
        Platform: {Platform.OS} {Platform.Version}
      </Text>
    </View>
  );
};

const styles = StyleSheet.create({
  card: {
    padding: 16,
    borderRadius: 12,
    backgroundColor: "#fff",
    margin: 8,
    // Platform-specific shadow
    ...Platform.select({
      ios: {
        shadowColor: "#000",
        shadowOffset: { width: 0, height: 2 },
        shadowOpacity: 0.1,
        shadowRadius: 8,
      },
      android: {
        elevation: 4,
      },
    }),
  },
  activeCard: {
    borderWidth: 2,
    borderColor: "#007AFF",
  },
  disabledCard: {
    opacity: 0.5,
  },
  text: {
    fontSize: 16,
    // Platform-specific font
    fontFamily: Platform.OS === "ios" ? "System" : "Roboto",
  },
  disabledText: {
    color: "#999",
  },
});
```

### 2.4 Shadows

    padding: 16,

      ios: {
        shadowColor: "#000",
        shadowOffset: { width: 0, height: 4 },

        shadowRadius: 12,

        elevation: 6,

      shadowColor: "#000",

      shadowRadius: 2,

## 3. Responsive Design

### 3.1 Dimensions API & useWindowDimensions

```tsx
import React from "react";
import {
  View,
  Text,
  StyleSheet,
  Dimensions,
  useWindowDimensions,
} from "react-native";

// ❌ Dimensions.get() — giá trị tĩnh, không update khi xoay màn hình
const { width: SCREEN_WIDTH, height: SCREEN_HEIGHT } = Dimensions.get("window");

const ResponsiveDemo = () => {
  // ✅ useWindowDimensions — reactive, update khi xoay
  const { width, height, fontScale } = useWindowDimensions();
  const isLandscape = width > height;

  return (
    <View style={styles.container}>
      <Text>
        Screen: {Math.round(width)} x {Math.round(height)}
      </Text>
      <Text>Orientation: {isLandscape ? "Landscape" : "Portrait"}</Text>
      <Text>Font Scale: {fontScale}</Text>

      {/* Responsive columns */}
      <View style={[styles.row, { flexWrap: "wrap" }]}>
        {[1, 2, 3, 4].map((i) => (
          <View
            key={i}
            style={[
              styles.card,
              {
                // 2 columns portrait, 4 columns landscape
                width: isLandscape ? width / 4 - 20 : width / 2 - 20,
              },
            ]}
          >
            <Text style={styles.cardText}>Card {i}</Text>
          </View>
        ))}
      </View>
    </View>
  );
};

const styles = StyleSheet.create({
  container: { flex: 1, padding: 12, gap: 12 },
  row: { flexDirection: "row", gap: 8 },
  card: {
    height: 100,
    backgroundColor: "#007AFF",
    borderRadius: 12,
    justifyContent: "center",
    alignItems: "center",
    marginBottom: 8,
  },
  cardText: { color: "#fff", fontWeight: "bold", fontSize: 16 },
});
```

### 3.2 Percentage Values & PixelRatio

```tsx
import React from "react";
import { View, Text, StyleSheet, PixelRatio } from "react-native";

const PercentAndPixelDemo = () => {
  // PixelRatio — tỷ lệ pixel của device
  const ratio = PixelRatio.get(); // 1, 2, 3, 3.5
  const fontScale = PixelRatio.getFontScale(); // User's font scale setting

  return (
    <View style={styles.container}>
      <Text>Pixel Ratio: {ratio}</Text>
      <Text>Font Scale: {fontScale}</Text>

      {/* Percentage widths */}
      <View style={styles.full}>
        <Text style={styles.label}>100%</Text>
      </View>
      <View style={styles.threeQuarter}>
        <Text style={styles.label}>75%</Text>
      </View>
      <View style={styles.half}>
        <Text style={styles.label}>50%</Text>
      </View>

      {/* Normalize font size based on pixel ratio */}
      <Text style={{ fontSize: normalize(16) }}>
        Normalized text (16dp → {normalize(16).toFixed(1)}px)
      </Text>
    </View>
  );
};

// Utility: Normalize size across devices
const normalize = (size: number): number => {
  const scale = PixelRatio.get();
  if (scale >= 3) return size; // High DPI
  if (scale >= 2) return size * 0.95; // Medium DPI
  return size * 0.85; // Low DPI
};

const styles = StyleSheet.create({
  container: { padding: 16, gap: 8 },
  full: {
    width: "100%",
    height: 40,
    backgroundColor: "#007AFF",
    justifyContent: "center",
    borderRadius: 4,
  },
  threeQuarter: {
    width: "75%",
    height: 40,
    backgroundColor: "#4ecdc4",
    justifyContent: "center",
    borderRadius: 4,
  },
  half: {
    width: "50%",
    height: 40,
    backgroundColor: "#ff6b6b",
    justifyContent: "center",
    borderRadius: 4,
  },
  label: { color: "#fff", fontWeight: "600", textAlign: "center" },
});
```

// exercises/ResponsiveCard.tsx
import React from "react";
import {
  View,
  Text,
  Image,
  StyleSheet,
  useWindowDimensions,
} from "react-native";

const ResponsiveCard = () => {
  const { width } = useWindowDimensions();
  const isTablet = width >= 768;
  const cardWidth = isTablet ? width / 3 - 24 : width - 32;

  return (
    <View style={styles.container}>
      {[1, 2, 3].map((i) => (

          />
          <View style={styles.cardBody}>
            <Text style={styles.cardTitle}>Card Title {i}</Text>
            <Text style={styles.cardDesc}>
              Responsive card adapts to screen size
            </Text>
          </View>
        </View>
      ))}
    </View>
  );
};

const styles = StyleSheet.create({
  container: {
    flexDirection: "row",
    flexWrap: "wrap",
    padding: 16,
    gap: 12,
    justifyContent: "center",
