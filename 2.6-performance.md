# 2.6 Performance Cơ Bản

> **Mục tiêu**: Nhận biết và tránh các vấn đề performance phổ biến trong React Native.

---

## 📖 Tổng Quan

Performance trong React = **giảm số lần và chi phí re-render không cần thiết**.

```
User thao tác → State thay đổi → Component re-render → UI cập nhật
                                       ↑
                              Đây là nơi cần optimize
```

---

## 1. Hiểu Re-render

### 1.1 Re-render là gì?

**Re-render** = React gọi lại function component để tính JSX mới, rồi so sánh với JSX cũ (diffing) để update UI.

```tsx
import React, { useState } from "react";
import { View, Text, TouchableOpacity, StyleSheet } from "react-native";

const ReRenderDemo = () => {
  const [count, setCount] = useState(0);

  // 🔄 Dòng này chạy LẠI mỗi khi component re-render
  console.log("🔄 ReRenderDemo rendered! Count:", count);

  // Tạo random number để chứng minh function chạy lại
  const randomValue = Math.random();

  return (
    <View style={styles.container}>
      <Text>Count: {count}</Text>
      <Text>Random (changes on re-render): {randomValue.toFixed(4)}</Text>
      <TouchableOpacity
        style={styles.btn}
        onPress={() => setCount((c) => c + 1)}
      >
        <Text style={styles.btnText}>Increment</Text>
      </TouchableOpacity>
    </View>
  );
};

const styles = StyleSheet.create({
  container: { padding: 16, gap: 8 },
  btn: { backgroundColor: "#007AFF", padding: 12, borderRadius: 8 },
  btnText: { color: "#fff", textAlign: "center", fontWeight: "600" },
});
```

### 1.2 Tại sao Re-render xảy ra?

Component re-render khi:

| Nguyên nhân                | Ví dụ                   |
| -------------------------- | ----------------------- |
| **State thay đổi**         | `setState(newValue)`    |
| **Props thay đổi**         | Parent truyền props mới |
| **Parent re-render**       | Dù props con không đổi! |
| **Context value thay đổi** | Provider value mới      |

```tsx
import React, { useState } from "react";
import { View, Text, TouchableOpacity, StyleSheet } from "react-native";

// 🔴 VẤN ĐỀ: ChildB re-render dù props không đổi
const Parent = () => {
  const [countA, setCountA] = useState(0);

  console.log("📦 Parent rendered");

  return (
    <View style={styles.container}>
      <TouchableOpacity
        style={styles.btn}
        onPress={() => setCountA((c) => c + 1)}
      >
        <Text style={styles.btnText}>Update Parent: {countA}</Text>
      </TouchableOpacity>
      <ChildA count={countA} />
      <ChildB name="Static" /> {/* ← Props không đổi, vẫn re-render! */}
    </View>
  );
};

const ChildA = ({ count }: { count: number }) => {
  console.log("  👶 ChildA rendered (count changed)");
  return <Text>ChildA: {count}</Text>;
};

const ChildB = ({ name }: { name: string }) => {
  console.log("  👶 ChildB rendered (UNNECESSARY! props unchanged)");
  return <Text>ChildB: {name}</Text>;
};

const styles = StyleSheet.create({
  container: { padding: 16, gap: 8 },
  btn: { backgroundColor: "#007AFF", padding: 12, borderRadius: 8 },
  btnText: { color: "#fff", textAlign: "center" },
});
```

## 2. React.memo() — Ngăn re-render khi props không đổi

`React.memo()` bọc component và **skip re-render** nếu props giống hệt lần trước.

```tsx
import React, { useState, memo } from "react";
import { View, Text, TouchableOpacity, StyleSheet } from "react-native";

const Parent = () => {
  const [countA, setCountA] = useState(0);

  console.log("📦 Parent rendered");

  return (
    <View style={styles.container}>
      <TouchableOpacity
        style={styles.btn}
        onPress={() => setCountA((c) => c + 1)}
      >
        <Text style={styles.btnText}>Update Parent: {countA}</Text>
      </TouchableOpacity>
      <ChildA count={countA} />
      <MemoizedChildB name="Static" /> {/* ✅ Không re-render! */}
    </View>
  );
};

const ChildA = ({ count }: { count: number }) => {
  console.log("  👶 ChildA rendered");
  return (
    <View style={styles.child}>
      <Text>ChildA: {count}</Text>
    </View>
  );
};

// ✅ Wrap với React.memo — chỉ re-render khi props thay đổi
const MemoizedChildB = memo(({ name }: { name: string }) => {
  console.log("  👶 MemoizedChildB rendered");
  return (
    <View style={styles.child}>
      <Text>ChildB (memo): {name}</Text>
    </View>
  );
});

// ✅ Cũng có thể dùng custom comparison
const MemoizedWithCompare = memo(
  ({ data }: { data: { id: number; value: string } }) => {
    console.log("  👶 Custom memo rendered");
    return <Text>{data.value}</Text>;
  },
  // Custom comparison: chỉ re-render khi id đổi
  (prevProps, nextProps) => prevProps.data.id === nextProps.data.id,
);

const styles = StyleSheet.create({
  container: { padding: 16, gap: 8 },
  btn: { backgroundColor: "#007AFF", padding: 12, borderRadius: 8 },
  btnText: { color: "#fff", textAlign: "center" },
  child: { padding: 12, backgroundColor: "#f0f0f0", borderRadius: 8 },
});
```

## 3. useMemo — Memoize Computed Values

Tránh tính toán lại những giá trị nặng mỗi render.

```tsx
import React, { useState, useMemo } from "react";
import { View, Text, TextInput, FlatList, StyleSheet } from "react-native";

const ExpensiveListDemo = () => {
  const [searchTerm, setSearchTerm] = useState("");
  const [selectedCategory, setSelectedCategory] = useState("all");
  const [darkMode, setDarkMode] = useState(false); // Unrelated state

  // Giả lập data lớn (1000 items)
  const allProducts = useMemo(() => {
    return Array.from({ length: 1000 }, (_, i) => ({
      id: String(i),
      name: `Product ${i}`,
      category: ["Electronics", "Books", "Clothing"][i % 3],
      price: Math.floor(Math.random() * 1000000),
    }));
  }, []); // Chỉ tạo 1 lần

  // ✅ useMemo — chỉ filter/sort lại khi searchTerm hoặc category đổi
  const filteredProducts = useMemo(() => {
    console.log("🔄 Filtering products..."); // Log để verify
    let result = allProducts;

    if (selectedCategory !== "all") {
      result = result.filter((p) => p.category === selectedCategory);
    }

    if (searchTerm) {
      result = result.filter((p) =>
        p.name.toLowerCase().includes(searchTerm.toLowerCase()),
      );
    }

    return result.sort((a, b) => a.price - b.price);
  }, [allProducts, searchTerm, selectedCategory]);
  // 👆 Không chạy lại khi darkMode đổi!

  const totalValue = useMemo(() => {
    return filteredProducts.reduce((sum, p) => sum + p.price, 0);
  }, [filteredProducts]);

  return (
    <View style={styles.container}>
      <TextInput
        style={styles.input}
        value={searchTerm}
        onChangeText={setSearchTerm}
        placeholder="Search products..."
      />
      <Text>
        {filteredProducts.length} products — Total:{" "}
        {totalValue.toLocaleString()}đ
      </Text>
      <FlatList
        data={filteredProducts}
        keyExtractor={(item) => item.id}
        renderItem={({ item }) => (
          <View style={styles.item}>
            <Text>{item.name}</Text>
            <Text style={styles.price}>{item.price.toLocaleString()}đ</Text>
          </View>
        )}
        initialNumToRender={20}
        maxToRenderPerBatch={10}
      />
    </View>
  );
};

const styles = StyleSheet.create({
  container: { flex: 1, padding: 16 },
  input: {
    borderWidth: 1,
    borderColor: "#ddd",
    borderRadius: 8,
    padding: 10,
    marginBottom: 8,
  },
  item: {
    flexDirection: "row",
    justifyContent: "space-between",
    padding: 10,
    borderBottomWidth: 1,
    borderBottomColor: "#eee",
  },
  price: { color: "#007AFF", fontWeight: "600" },
});
```

## 4. useCallback — Memoize Functions

Tránh tạo function mới mỗi render — quan trọng khi truyền function làm prop cho `memo` component.

```tsx
import React, { useState, useCallback, memo } from "react";
import {
  View,
  Text,
  TouchableOpacity,
  FlatList,
  StyleSheet,
} from "react-native";

interface ItemProps {
  id: string;
  title: string;
  onPress: (id: string) => void;
  onDelete: (id: string) => void;
}

// ✅ Memo component — chỉ re-render khi props thay đổi
const ListItem = memo(({ id, title, onPress, onDelete }: ItemProps) => {
  console.log(`  Rendering item: ${title}`);
  return (
    <View style={styles.item}>
      <TouchableOpacity onPress={() => onPress(id)} style={{ flex: 1 }}>
        <Text>{title}</Text>
      </TouchableOpacity>
      <TouchableOpacity onPress={() => onDelete(id)}>
        <Text>🗑</Text>
      </TouchableOpacity>
    </View>
  );
});

const OptimizedList = () => {
  const [items, setItems] = useState(
    Array.from({ length: 50 }, (_, i) => ({
      id: String(i),
      title: `Item ${i}`,
    })),
  );
  const [selectedId, setSelectedId] = useState<string | null>(null);

  // ✅ useCallback — function reference ổn định
  const handlePress = useCallback((id: string) => {
    setSelectedId(id);
  }, []);

  const handleDelete = useCallback((id: string) => {
    setItems((prev) => prev.filter((item) => item.id !== id));
  }, []);

  // ❌ Nếu không dùng useCallback:
  // const handlePress = (id: string) => setSelectedId(id);
  // → Tạo function MỚI mỗi render → tất cả ListItem re-render!

  return (
    <View style={styles.container}>
      <Text style={styles.title}>Selected: {selectedId ?? "None"}</Text>
      <FlatList
        data={items}
        keyExtractor={(item) => item.id}
        renderItem={({ item }) => (
          <ListItem
            id={item.id}
            title={item.title}
            onPress={handlePress}
            onDelete={handleDelete}
          />
        )}
      />
    </View>
  );
};

const styles = StyleSheet.create({
  container: { flex: 1, padding: 16 },
  title: { fontSize: 18, fontWeight: "bold", marginBottom: 12 },
  item: {
    flexDirection: "row",
    alignItems: "center",
    padding: 12,
    backgroundColor: "#f9f9f9",
    marginBottom: 4,
    borderRadius: 6,
  },
});
```

## 5. Key Prop trong Lists

`key` không chỉ tránh warning — nó ảnh hưởng **trực tiếp** đến performance.

```tsx
import React, { useState } from "react";
import {
  View,
  Text,
  TouchableOpacity,
  TextInput,
  StyleSheet,
} from "react-native";

interface Todo {
  id: string;
  text: string;
}

const KeyPropDemo = () => {
  const [todos, setTodos] = useState<Todo[]>([
    { id: "a", text: "Learn React" },
    { id: "b", text: "Learn RN" },
    { id: "c", text: "Build App" },
  ]);

  const addToTop = () => {
    setTodos((prev) => [
      { id: Date.now().toString(), text: `New todo ${prev.length + 1}` },
      ...prev,
    ]);
  };

  return (
    <View style={styles.container}>
      <TouchableOpacity style={styles.btn} onPress={addToTop}>
        <Text style={styles.btnText}>Add to Top</Text>
      </TouchableOpacity>

      {/* ✅ ĐÚNG — dùng unique ID */}
      {todos.map((todo) => (
        <View key={todo.id} style={styles.item}>
          <TextInput defaultValue={todo.text} style={styles.input} />
        </View>
      ))}

      {/* ❌ SAI — dùng index
         Khi thêm item vào đầu, tất cả index shift → React nghĩ MỌI item đổi
         → TextInput mất focus, state lẫn lộn */}
      {/* {todos.map((todo, index) => (
        <View key={index} style={styles.item}>
          <TextInput defaultValue={todo.text} style={styles.input} />
        </View>
      ))} */}
    </View>
  );
};

const styles = StyleSheet.create({
  container: { padding: 16, gap: 8 },
  btn: { backgroundColor: "#007AFF", padding: 12, borderRadius: 8 },
  btnText: { color: "#fff", textAlign: "center", fontWeight: "600" },
  item: { backgroundColor: "#f0f0f0", borderRadius: 8 },
  input: { padding: 12 },
});
```

## 6. Tránh Inline Objects/Functions

```tsx
import React, { useState, useCallback, useMemo, memo } from "react";
import { View, Text, TouchableOpacity, StyleSheet } from "react-native";

const ChildComponent = memo(
  ({
    style,
    onPress,
    config,
  }: {
    style: object;
    onPress: () => void;
    config: { color: string };
  }) => {
    console.log("Child rendered");
    return (
      <TouchableOpacity style={style} onPress={onPress}>
        <Text style={{ color: config.color }}>Click me</Text>
      </TouchableOpacity>
    );
  },
);

const InlineAvoidanceDemo = () => {
  const [count, setCount] = useState(0);

  // ❌ SAI — Inline objects tạo reference mới mỗi render
  // → memo vô tác dụng vì props luôn "khác"
  // <ChildComponent
  //   style={{ padding: 12, backgroundColor: '#007AFF' }}  ← NEW object mỗi render
  //   onPress={() => console.log('pressed')}                ← NEW function mỗi render
  //   config={{ color: 'white' }}                            ← NEW object mỗi render
  // />

  // ✅ ĐÚNG — Dùng useMemo/useCallback/StyleSheet
  const buttonStyle = useMemo(
    () => ({
      padding: 12,
      backgroundColor: "#007AFF",
      borderRadius: 8,
    }),
    [],
  );

  const handlePress = useCallback(() => {
    console.log("pressed");
  }, []);

  const config = useMemo(() => ({ color: "white" }), []);

  return (
    <View style={styles.container}>
      <Text>Count: {count}</Text>
      <TouchableOpacity
        style={styles.btn}
        onPress={() => setCount((c) => c + 1)}
      >
        <Text style={styles.btnText}>Increment (parent re-render)</Text>
      </TouchableOpacity>

      {/* ✅ Child sẽ KHÔNG re-render khi count đổi */}
      <ChildComponent
        style={buttonStyle}
        onPress={handlePress}
        config={config}
      />
    </View>
  );
};

const styles = StyleSheet.create({
  container: { padding: 16, gap: 12 },
  btn: { backgroundColor: "#6c757d", padding: 12, borderRadius: 8 },
  btnText: { color: "#fff", textAlign: "center" },
});
```

## 7. Cheat Sheet: Khi Nào Dùng Gì?

```
Muốn ngăn child component re-render?
│
├─ Props là primitive (string, number)?
│   └─ ✅ React.memo() là đủ
│
├─ Props có function?
│   └─ ✅ React.memo() + useCallback cho function
│
├─ Props có object/array?
│   └─ ✅ React.memo() + useMemo cho object/array
│
└─ Computed value nặng?
    └─ ✅ useMemo cho giá trị tính toán
```

⚠️ **Đừng optimize quá sớm (premature optimization)**:

- Chỉ optimize khi thấy **lag thực sự**
- `React.memo`, `useMemo`, `useCallback` cũng có **cost** (memory)
- Measure trước, optimize sau

## 🏋️ Bài Tập Thực Hành

### Bài 1: Re-render Logger

```tsx
// exercises/RerenderLogger.tsx
import React, { useState, useRef, useEffect, memo } from "react";
import { View, Text, TouchableOpacity, StyleSheet } from "react-native";

// Component log re-render count
const RenderCounter = memo(({ label }: { label: string }) => {
  const renderCount = useRef(0);
  renderCount.current += 1;

  return (
    <View style={styles.counter}>
      <Text>{label}</Text>
      <Text style={styles.count}>Renders: {renderCount.current}</Text>
    </View>
  );
});

const UnoptimizedChild = ({ label }: { label: string }) => {
  const renderCount = useRef(0);
  renderCount.current += 1;

  return (
    <View style={styles.counter}>
      <Text>{label}</Text>
      <Text style={[styles.count, { color: "#dc3545" }]}>
        Renders: {renderCount.current}
      </Text>
    </View>
  );
};

const RerenderLoggerExercise = () => {
  const [count, setCount] = useState(0);
  const [text, setText] = useState("");

  return (
    <View style={styles.container}>
      <Text style={styles.title}>Re-render Logger</Text>

      <TouchableOpacity
        style={styles.btn}
        onPress={() => setCount((c) => c + 1)}
      >
        <Text style={styles.btnText}>Count: {count}</Text>
      </TouchableOpacity>

      <Text style={styles.label}>❌ Without memo (re-renders every time):</Text>
      <UnoptimizedChild label="Unoptimized" />

      <Text style={styles.label}>✅ With memo (stable):</Text>
      <RenderCounter label="Optimized" />
    </View>
  );
};

const styles = StyleSheet.create({
  container: { padding: 16, gap: 8 },
  title: { fontSize: 22, fontWeight: "bold", marginBottom: 12 },
  label: { fontSize: 13, fontWeight: "600", color: "#666", marginTop: 12 },
  btn: { backgroundColor: "#007AFF", padding: 12, borderRadius: 8 },
  btnText: { color: "#fff", textAlign: "center", fontWeight: "600" },
  counter: {
    flexDirection: "row",
    justifyContent: "space-between",
    padding: 12,
    backgroundColor: "#f9f9f9",
    borderRadius: 8,
  },
  count: { fontWeight: "bold", color: "#4CAF50" },
});

export default RerenderLoggerExercise;
```
