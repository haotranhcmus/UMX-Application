# 3.10 Local Storage

> **Mục tiêu**: Lưu trữ dữ liệu local trên device — AsyncStorage, SecureStore, MMKV, và SQLite.

---

## 📖 Tổng Quan

| Giải pháp        | Use case                          | Encrypted | Speed         | Data size   |
| ---------------- | --------------------------------- | --------- | ------------- | ----------- |
| **AsyncStorage** | Settings, cache, simple data      | ❌        | Chậm          | < 6 MB      |
| **SecureStore**  | Tokens, passwords, secrets        | ✅        | Trung bình    | < 2 KB/item |
| **MMKV**         | Thay thế AsyncStorage (nhanh hơn) | Optional  | **Rất nhanh** | Lớn         |
| **SQLite**       | Structured data, offline DB       | ❌        | Nhanh         | Lớn         |

## 1. AsyncStorage

### 1.1 Cài Đặt

```bash
npx expo install @react-native-async-storage/async-storage
```

### 1.2 CRUD Operations

```tsx
// services/storage.ts
import AsyncStorage from '@react-native-async-storage/async-storage';

// ===== Basic Operations =====

// SET — lưu string
export const setItem = async (key: string, value: string): Promise<void> => {
  try {
    await AsyncStorage.setItem(key, value);
  } catch (error) {
    console.error('AsyncStorage setItem error:', error);
  }
};

// GET — đọc string
export const getItem = async (key: string): Promise<string | null> => {
  try {
    return await AsyncStorage.getItem(key);
  } catch (error) {
    console.error('AsyncStorage getItem error:', error);
    return null;
  }
};

// REMOVE — xóa 1 key
export const removeItem = async (key: string): Promise<void> => {
  try {
    await AsyncStorage.removeItem(key);
  } catch (error) {
    console.error('AsyncStorage removeItem error:', error);
  }
};

// ===== JSON Operations =====

// SET Object/Array → tự stringify
export const setObject = async <T>(key: string, value: T): Promise<void> => {
  try {
    const jsonValue = JSON.stringify(value);
    await AsyncStorage.setItem(key, jsonValue);
  } catch (error) {
    console.error('AsyncStorage setObject error:', error);
  }
};

// GET Object/Array → tự parse
export const getObject = async <T>(key: string): Promise<T | null> => {
  try {
    const jsonValue = await AsyncStorage.getItem(key);
    return jsonValue != null ? JSON.parse(jsonValue) : null;
  } catch (error) {
    console.error('AsyncStorage getObject error:', error);
    return null;
  }
};

// ===== Multi Operations =====

// SET nhiều key cùng lúc
export const setMultiple = async (items: [string, string][]): Promise<void> => {
  try {
    await AsyncStorage.multiSet(items);
  } catch (error) {
    console.error('AsyncStorage multiSet error:', error);
  }
};

// GET nhiều key cùng lúc
export const getMultiple = async (keys: string[]): Promise<readonly [string, string | null][]> => {
  try {
    return await AsyncStorage.multiGet(keys);
  } catch (error) {
    console.error('AsyncStorage multiGet error:', error);
    return [];
  }
};

// REMOVE nhiều keys
export const removeMultiple = async (keys: string[]): Promise<void> => {
  try {
    await AsyncStorage.multiRemove(keys);
  } catch (error) {
    console.error('AsyncStorage multiRemove error:', error);
  }
};

// GET tất cả keys
export const getAllKeys = async (): Promise<readonly string[]> => {
  try {
    return await AsyncStorage.getAllKeys();
  } catch (error) {
    console.error('AsyncStorage getAllKeys error:', error);
    return [];
  }
};

// CLEAR tất cả — ⚠️ Cẩn thận!
export const clearAll = async (): Promise<void> => {
  try {
    await AsyncStorage.clear();
  } catch (error) {
    console.error('AsyncStorage clear error:', error);
  }
};
```

### 1.3 Sử Dụng Trong Component

```tsx
import React, { useState, useEffect } from "react";
import {
  View,
  Text,
  TextInput,
  TouchableOpacity,
  StyleSheet,
  Alert,
  Switch,
} from "react-native";
import { setObject, getObject } from "../services/storage";

interface UserPreferences {
  theme: "light" | "dark";
  fontSize: number;
  notificationsEnabled: boolean;
  language: string;
}

const PREFS_KEY = "@user_preferences";

const defaultPrefs: UserPreferences = {
  theme: "light",
  fontSize: 16,
  notificationsEnabled: true,
  language: "vi",
};

const SettingsScreen = () => {
  const [prefs, setPrefs] = useState<UserPreferences>(defaultPrefs);
  const [loaded, setLoaded] = useState(false);

  // Load preferences khi mount
  useEffect(() => {
    const loadPrefs = async () => {
      const stored = await getObject<UserPreferences>(PREFS_KEY);
      if (stored) {
        setPrefs(stored);
      }
      setLoaded(true);
    };
    loadPrefs();
  }, []);

  // Save mỗi khi prefs thay đổi
  useEffect(() => {
    if (loaded) {
      setObject(PREFS_KEY, prefs);
    }
  }, [prefs, loaded]);

  const updatePref = <K extends keyof UserPreferences>(
    key: K,
    value: UserPreferences[K],
  ) => {
    setPrefs((prev) => ({ ...prev, [key]: value }));
  };

  return (
    <View style={sStyles.container}>
      <Text style={sStyles.title}>⚙️ Settings</Text>

      {/* Theme Toggle */}
      <View style={sStyles.row}>
        <Text style={sStyles.label}>Dark Mode</Text>
        <Switch
          value={prefs.theme === "dark"}
          onValueChange={(v) => updatePref("theme", v ? "dark" : "light")}
        />
      </View>

      {/* Notifications */}
      <View style={sStyles.row}>
        <Text style={sStyles.label}>Notifications</Text>
        <Switch
          value={prefs.notificationsEnabled}
          onValueChange={(v) => updatePref("notificationsEnabled", v)}
        />
      </View>

      {/* Font Size */}
      <View style={sStyles.row}>
        <Text style={sStyles.label}>Font Size: {prefs.fontSize}</Text>
        <View style={{ flexDirection: "row", gap: 8 }}>
          <TouchableOpacity
            style={sStyles.sizeBtn}
            onPress={() =>
              updatePref("fontSize", Math.max(12, prefs.fontSize - 2))
            }
          >
            <Text>A−</Text>
          </TouchableOpacity>
          <TouchableOpacity
            style={sStyles.sizeBtn}
            onPress={() =>
              updatePref("fontSize", Math.min(24, prefs.fontSize + 2))
            }
          >
            <Text>A+</Text>
          </TouchableOpacity>
        </View>
      </View>

      {/* Preview */}
      <Text style={[sStyles.preview, { fontSize: prefs.fontSize }]}>
        Preview text with current font size ({prefs.fontSize}px)
      </Text>
    </View>
  );
};

const sStyles = StyleSheet.create({
  container: { flex: 1, padding: 20, backgroundColor: "#fff" },
  title: { fontSize: 24, fontWeight: "bold", marginBottom: 24 },
  row: {
    flexDirection: "row",
    justifyContent: "space-between",
    alignItems: "center",
    paddingVertical: 16,
    borderBottomWidth: 1,
    borderBottomColor: "#f0f0f0",
  },
  label: { fontSize: 16, color: "#333" },
  sizeBtn: {
    width: 40,
    height: 40,
    borderWidth: 1,
    borderColor: "#ddd",
    borderRadius: 8,
    justifyContent: "center",
    alignItems: "center",
  },
  preview: { marginTop: 24, color: "#666", lineHeight: 28 },
});

export default SettingsScreen;
```

## 2. Custom Hook — useAsyncStorage

```tsx
// hooks/useAsyncStorage.ts
import { useState, useEffect, useCallback } from "react";
import AsyncStorage from "@react-native-async-storage/async-storage";

export function useAsyncStorage<T>(key: string, initialValue: T) {
  const [storedValue, setStoredValue] = useState<T>(initialValue);
  const [loading, setLoading] = useState(true);

  // Load từ storage
  useEffect(() => {
    const load = async () => {
      try {
        const item = await AsyncStorage.getItem(key);
        if (item !== null) {
          setStoredValue(JSON.parse(item));
        }
      } catch (error) {
        console.error(`useAsyncStorage load error [${key}]:`, error);
      } finally {
        setLoading(false);
      }
    };
    load();
  }, [key]);

  // Save function
  const setValue = useCallback(
    async (value: T | ((prev: T) => T)) => {
      try {
        const valueToStore =
          value instanceof Function ? value(storedValue) : value;
        setStoredValue(valueToStore);
        await AsyncStorage.setItem(key, JSON.stringify(valueToStore));
      } catch (error) {
        console.error(`useAsyncStorage save error [${key}]:`, error);
      }
    },
    [key, storedValue],
  );

  // Remove function
  const removeValue = useCallback(async () => {
    try {
      await AsyncStorage.removeItem(key);
      setStoredValue(initialValue);
    } catch (error) {
      console.error(`useAsyncStorage remove error [${key}]:`, error);
    }
  }, [key, initialValue]);

  return { value: storedValue, setValue, removeValue, loading };
}

// ===== Sử dụng =====
// const { value: theme, setValue: setTheme, loading } = useAsyncStorage<'light' | 'dark'>('@theme', 'light');
// const { value: bookmarks, setValue: setBookmarks } = useAsyncStorage<string[]>('@bookmarks', []);
```

## 3. Expo SecureStore

Xem chi tiết ở [3.9 Authentication Flow](./3.9-authentication-flow.md). Tóm tắt:

```tsx
import * as SecureStore from "expo-secure-store";

// Lưu (encrypted)
await SecureStore.setItemAsync("token", "my-secret-token");

// Đọc
const token = await SecureStore.getItemAsync("token");

// Xóa
await SecureStore.deleteItemAsync("token");
```

**Lưu ý:**

- Giới hạn **2048 bytes** per item
- Chỉ lưu **string** (không lưu object trực tiếp)
- Dùng cho: tokens, passwords, API keys
- **Không** dùng cho: data lớn, settings, cache

## 4. MMKV (Nhanh hơn AsyncStorage 30x)

### 4.1 Cài Đặt

```bash
npx expo install react-native-mmkv
```

### 4.2 Setup & Sử Dụng

```tsx
// services/mmkvStorage.ts
import { MMKV } from 'react-native-mmkv';

// Tạo instance
export const storage = new MMKV({
  id: 'app-storage',
  // encryptionKey: 'your-encryption-key', // Optional encryption
});

// ===== Wrapper với TypeScript =====
export const mmkvStorage = {
  // String
  getString: (key: string): string | undefined => storage.getString(key),
  setString: (key: string, value: string): void => storage.set(key, value),

  // Number
  getNumber: (key: string): number | undefined => storage.getNumber(key),
  setNumber: (key: string, value: number): void => storage.set(key, value),

  // Boolean
  getBoolean: (key: string): boolean | undefined => storage.getBoolean(key),
  setBoolean: (key: string, value: boolean): void => storage.set(key, value),

  // Object/Array — JSON
  getObject: <T>(key: string): T | null => {
    const json = storage.getString(key);
    return json ? JSON.parse(json) : null;
  },
  setObject: <T>(key: string, value: T): void => {
    storage.set(key, JSON.stringify(value));
  },

  // Delete
  delete: (key: string): void => storage.delete(key),

  // Has key
  contains: (key: string): boolean => storage.contains(key),

  // Get all keys
  getAllKeys: (): string[] => storage.getAllKeys(),

  // Clear all
  clearAll: (): void => storage.clearAll(),
};

// ===== Sử dụng =====
// mmkvStorage.setString('username', 'john');
// const username = mmkvStorage.getString('username');
//
// mmkvStorage.setObject('user', { id: 1, name: 'John' });
// const user = mmkvStorage.getObject<{ id: number; name: string }>('user');
```

### 4.3 MMKV + Zustand Persist

```tsx
// stores/useSettingsStore.ts
import { create } from "zustand";
import { persist, createJSONStorage, StateStorage } from "zustand/middleware";
import { storage } from "../services/mmkvStorage";

// Adapter cho Zustand
const mmkvZustandStorage: StateStorage = {
  getItem: (name) => storage.getString(name) ?? null,
  setItem: (name, value) => storage.set(name, value),
  removeItem: (name) => storage.delete(name),
};

interface SettingsState {
  theme: "light" | "dark";
  language: "vi" | "en";
  setTheme: (theme: "light" | "dark") => void;
  setLanguage: (lang: "vi" | "en") => void;
}

export const useSettingsStore = create<SettingsState>()(
  persist(
    (set) => ({
      theme: "light",
      language: "vi",
      setTheme: (theme) => set({ theme }),
      setLanguage: (language) => set({ language }),
    }),
    {
      name: "settings",
      storage: createJSONStorage(() => mmkvZustandStorage),
    },
  ),
);
```

## 5. SQLite (Expo)

Cho dữ liệu structured, offline-first, quantities lớn.

### 5.1 Cài Đặt

```bash
npx expo install expo-sqlite
```

### 5.2 Database Service

```tsx
// services/database.ts
import * as SQLite from "expo-sqlite";

// Mở / tạo database
const db = SQLite.openDatabaseSync("myapp.db");

// ===== Init tables =====
export const initDatabase = () => {
  db.execSync(`
    CREATE TABLE IF NOT EXISTS notes (
      id INTEGER PRIMARY KEY AUTOINCREMENT,
      title TEXT NOT NULL,
      content TEXT,
      created_at TEXT DEFAULT (datetime('now')),
      updated_at TEXT DEFAULT (datetime('now'))
    );
  `);
};

// ===== CRUD =====
export interface Note {
  id: number;
  title: string;
  content: string;
  created_at: string;
  updated_at: string;
}

// CREATE
export const createNote = (title: string, content: string): Note => {
  const result = db.runSync(
    "INSERT INTO notes (title, content) VALUES (?, ?)",
    [title, content],
  );
  return getNote(result.lastInsertRowId) as Note;
};

// READ — all
export const getNotes = (): Note[] => {
  return db.getAllSync<Note>("SELECT * FROM notes ORDER BY updated_at DESC");
};

// READ — single
export const getNote = (id: number): Note | null => {
  return db.getFirstSync<Note>("SELECT * FROM notes WHERE id = ?", [id]);
};

// UPDATE
export const updateNote = (
  id: number,
  title: string,
  content: string,
): void => {
  db.runSync(
    "UPDATE notes SET title = ?, content = ?, updated_at = datetime('now') WHERE id = ?",
    [title, content, id],
  );
};

// DELETE
export const deleteNote = (id: number): void => {
  db.runSync("DELETE FROM notes WHERE id = ?", [id]);
};

// SEARCH
export const searchNotes = (query: string): Note[] => {
  return db.getAllSync<Note>(
    "SELECT * FROM notes WHERE title LIKE ? OR content LIKE ? ORDER BY updated_at DESC",
    [`%${query}%`, `%${query}%`],
  );
};
```

### 5.3 Sử dụng trong Component

```tsx
// screens/NotesScreen.tsx
import React, { useState, useEffect, useCallback } from "react";
import {
  View,
  Text,
  FlatList,
  TouchableOpacity,
  TextInput,
  StyleSheet,
  Alert,
} from "react-native";
import {
  initDatabase,
  getNotes,
  createNote,
  deleteNote,
  Note,
} from "../services/database";

const NotesScreen = () => {
  const [notes, setNotes] = useState<Note[]>([]);
  const [title, setTitle] = useState("");

  useEffect(() => {
    initDatabase();
    loadNotes();
  }, []);

  const loadNotes = () => {
    const data = getNotes();
    setNotes(data);
  };

  const handleAdd = () => {
    if (title.trim()) {
      createNote(title.trim(), "");
      setTitle("");
      loadNotes();
    }
  };

  const handleDelete = (id: number) => {
    Alert.alert("Xóa", "Bạn có chắc?", [
      { text: "Hủy", style: "cancel" },
      {
        text: "Xóa",
        style: "destructive",
        onPress: () => {
          deleteNote(id);
          loadNotes();
        },
      },
    ]);
  };

  return (
    <View style={nStyles.container}>
      <View style={nStyles.inputRow}>
        <TextInput
          style={nStyles.input}
          value={title}
          onChangeText={setTitle}
          placeholder="New note..."
          onSubmitEditing={handleAdd}
        />
        <TouchableOpacity style={nStyles.addBtn} onPress={handleAdd}>
          <Text style={nStyles.addBtnText}>+</Text>
        </TouchableOpacity>
      </View>

      <FlatList
        data={notes}
        keyExtractor={(item) => item.id.toString()}
        renderItem={({ item }) => (
          <View style={nStyles.noteItem}>
            <View style={{ flex: 1 }}>
              <Text style={nStyles.noteTitle}>{item.title}</Text>
              <Text style={nStyles.noteDate}>{item.created_at}</Text>
            </View>
            <TouchableOpacity onPress={() => handleDelete(item.id)}>
              <Text style={{ fontSize: 18 }}>🗑️</Text>
            </TouchableOpacity>
          </View>
        )}
        ListEmptyComponent={<Text style={nStyles.empty}>No notes yet</Text>}
      />
    </View>
  );
};

const nStyles = StyleSheet.create({

    padding: 12,
    fontSize: 16,
  },

    width: 48,
    backgroundColor: "#007AFF",
    borderRadius: 8,
    justifyContent: "center",
    alignItems: "center",
  },
  addBtnText: { color: "#fff", fontSize: 24 },
```

## 6. Chọn Giải Pháp Nào?

```
Bạn cần lưu gì?
│
├── Tokens, passwords, secrets?
│   └── ✅ SecureStore
│
├── Settings, preferences, simple cache?
│   ├── Cần tốc độ cao → ✅ MMKV
│   └── Đơn giản, ít data → ✅ AsyncStorage
│
├── Structured data, queries, offline DB?
│   └── ✅ SQLite
│
└── State persist (Zustand/Redux)?
    ├── ✅ MMKV + Zustand persist (fastest)
    └── ✅ AsyncStorage + Zustand persist
```
