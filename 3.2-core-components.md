# 3.2 Core Components

> **Mục tiêu**: Thành thạo các components cơ bản của React Native.

---

## 📖 Tổng Quan

React Native cung cấp các **core components** map trực tiếp sang native UI:

| React Native   | Android             | iOS            | Web (tương đương)  |
| -------------- | ------------------- | -------------- | ------------------ |
| `<View>`       | `android.view.View` | `UIView`       | `<div>`            |
| `<Text>`       | `TextView`          | `UITextView`   | `<p>`              |
| `<Image>`      | `ImageView`         | `UIImageView`  | `<img>`            |
| `<TextInput>`  | `EditText`          | `UITextField`  | `<input>`          |
| `<ScrollView>` | `ScrollView`        | `UIScrollView` | `<div>` + overflow |
| `<FlatList>`   | `RecyclerView`      | `UITableView`  | virtualized list   |

## 1. Basic Components

### 1.1 View — Container Component

`View` là building block cơ bản nhất — tương đương `<div>` trong web.

```tsx
import React from "react";
import { View, StyleSheet } from "react-native";

const ViewDemo = () => {
  return (
    // View là container — không hiển thị gì, chỉ layout
    <View style={styles.container}>
      <View style={styles.box1} />
      <View style={styles.box2} />
      <View style={styles.box3} />
    </View>
  );
};

const styles = StyleSheet.create({
  container: {
    flex: 1,
    flexDirection: "row", // Horizontal layout
    justifyContent: "center", // Center theo main axis
    alignItems: "center", // Center theo cross axis
    gap: 12,
  },
  box1: { width: 60, height: 60, backgroundColor: "#ff6b6b", borderRadius: 8 },
  box2: { width: 60, height: 60, backgroundColor: "#4ecdc4", borderRadius: 8 },
  box3: { width: 60, height: 60, backgroundColor: "#45b7d1", borderRadius: 8 },
});

export default ViewDemo;
```

## 2. List Components

### 2.1 FlatList — Optimized List

`FlatList` chỉ render items **đang hiển thị trên màn hình** → hiệu suất tốt với list lớn.

```tsx
import React, { useState, useCallback } from "react";
import {
  View,
  Text,
  FlatList,
  StyleSheet,
  Image,
  TouchableOpacity,
  RefreshControl,
  ActivityIndicator,
} from "react-native";

interface Contact {
  id: string;
  name: string;
  phone: string;
  avatar: string;
}

const ContactList = () => {
  const [contacts] = useState<Contact[]>(
    Array.from({ length: 100 }, (_, i) => ({
      id: String(i),
      name: `Contact ${i + 1}`,
      phone: `0${Math.floor(100000000 + Math.random() * 900000000)}`,
      avatar: `https://i.pravatar.cc/50?img=${(i % 70) + 1}`,
    })),
  );
  const [refreshing, setRefreshing] = useState(false);

  const onRefresh = useCallback(async () => {
    setRefreshing(true);
    await new Promise((res) => setTimeout(res, 1500));
    setRefreshing(false);
  }, []);

  // ✅ renderItem nên dùng useCallback hoặc khai báo ngoài
  const renderItem = useCallback(
    ({ item }: { item: Contact }) => (
      <TouchableOpacity style={styles.contactRow}>
        <Image source={{ uri: item.avatar }} style={styles.avatar} />
        <View style={styles.contactInfo}>
          <Text style={styles.contactName}>{item.name}</Text>
          <Text style={styles.contactPhone}>{item.phone}</Text>
        </View>
      </TouchableOpacity>
    ),
    [],
  );

  // ✅ keyExtractor
  const keyExtractor = useCallback((item: Contact) => item.id, []);

  // ✅ Separator
  const ItemSeparator = () => <View style={styles.separator} />;

  // ✅ Empty list
  const ListEmpty = () => (
    <View style={styles.empty}>
      <Text style={styles.emptyText}>📭 Không có liên hệ</Text>
    </View>
  );

  // ✅ Header & Footer
  const ListHeader = () => (
    <Text style={styles.header}>📇 Danh bạ ({contacts.length})</Text>
  );

  const ListFooter = () => <Text style={styles.footer}>— Hết danh sách —</Text>;

  return (
    <FlatList
      data={contacts}
      renderItem={renderItem}
      keyExtractor={keyExtractor}
      ItemSeparatorComponent={ItemSeparator}
      ListEmptyComponent={ListEmpty}
      ListHeaderComponent={ListHeader}
      ListFooterComponent={ListFooter}
      // Pull to refresh
      refreshControl={
        <RefreshControl refreshing={refreshing} onRefresh={onRefresh} />
      }
      // Performance optimization
      initialNumToRender={15} // Render 15 items đầu
      maxToRenderPerBatch={10} // Render 10 items mỗi batch
      windowSize={5} // Giữ 5 screens trong memory
      removeClippedSubviews={true} // Android: remove items ngoài view
      getItemLayout={(_, index) => ({
        // Biết trước height → skip measure
        length: 72,
        offset: 72 * index,
        index,
      })}
    />
  );
};

const styles = StyleSheet.create({
  contactRow: {
    flexDirection: "row",
    alignItems: "center",
    padding: 12,
    height: 72,
  },
  avatar: { width: 48, height: 48, borderRadius: 24, backgroundColor: "#eee" },
  contactInfo: { marginLeft: 12, flex: 1 },
  contactName: { fontSize: 16, fontWeight: "600" },
  contactPhone: { fontSize: 14, color: "#666", marginTop: 2 },
  separator: { height: 1, backgroundColor: "#f0f0f0", marginLeft: 72 },
  empty: { padding: 40, alignItems: "center" },
  emptyText: { fontSize: 16, color: "#999" },
  header: {
    fontSize: 20,
    fontWeight: "bold",
    padding: 16,
    backgroundColor: "#f8f9fa",
  },
  footer: { padding: 20, textAlign: "center", color: "#999" },
});

export default ContactList;
```

### 2.3 Infinite Scroll / Load More

```tsx
import React, { useState, useCallback } from "react";
import {
  View,
  Text,
  FlatList,
  ActivityIndicator,
  StyleSheet,
} from "react-native";

const InfiniteScrollDemo = () => {
  const [data, setData] = useState(generateItems(0, 20));
  const [loading, setLoading] = useState(false);
  const [page, setPage] = useState(1);
  const [hasMore, setHasMore] = useState(true);

  const loadMore = useCallback(async () => {
    if (loading || !hasMore) return;

    setLoading(true);
    // Giả lập API call
    await new Promise((res) => setTimeout(res, 1000));

    const newItems = generateItems(page * 20, 20);
    if (newItems.length === 0) {
      setHasMore(false);
    } else {
      setData((prev) => [...prev, ...newItems]);
      setPage((p) => p + 1);
    }
    setLoading(false);
  }, [loading, hasMore, page]);

  const renderFooter = () => {
    if (!loading) return null;
    return (
      <View style={styles.footer}>
        <ActivityIndicator size="small" color="#007AFF" />
        <Text style={styles.footerText}>Đang tải thêm...</Text>
      </View>
    );
  };

  return (
    <FlatList
      data={data}
      keyExtractor={(item) => item.id}
      renderItem={({ item }) => (
        <View style={styles.item}>
          <Text style={styles.itemText}>{item.title}</Text>
        </View>
      )}
      onEndReached={loadMore} // Gọi khi gần cuối list
      onEndReachedThreshold={0.5} // 50% trước khi chạm cuối
      ListFooterComponent={renderFooter}
    />
  );
};

function generateItems(start: number, count: number) {
  return Array.from({ length: count }, (_, i) => ({
    id: String(start + i),
    title: `Item #${start + i + 1}`,
  }));
}

const styles = StyleSheet.create({
  item: { padding: 16, borderBottomWidth: 1, borderBottomColor: "#eee" },
  itemText: { fontSize: 16 },
  footer: {
    padding: 16,
    alignItems: "center",
    flexDirection: "row",
    justifyContent: "center",
    gap: 8,
  },
  footerText: { color: "#666" },
});
```
