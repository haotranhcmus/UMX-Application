# 2.4 Component Lifecycle

> **Mục tiêu**: Hiểu lifecycle (vòng đời) của function component qua hooks.

---

## 📖 Tổng Quan

Mỗi component trong React có **3 giai đoạn**:

```
┌─────────────┐     ┌──────────────┐     ┌──────────────┐
│   MOUNT     │ ──► │   UPDATE     │ ──► │   UNMOUNT    │
│ (Tạo mới)   │     │ (Cập nhật)   │     │ (Xóa bỏ)    │
└─────────────┘     └──────────────┘     └──────────────┘
                         ↑    │
                         └────┘ (lặp lại khi state/props đổi)
```

| Giai đoạn   | Khi nào                       | useEffect tương ứng              |
| ----------- | ----------------------------- | -------------------------------- |
| **Mount**   | Component được render lần đầu | `useEffect(() => {...}, [])`     |
| **Update**  | State hoặc props thay đổi     | `useEffect(() => {...}, [deps])` |
| **Unmount** | Component bị remove khỏi UI   | `return () => { cleanup }`       |

---

## 1. Mount Phase — `useEffect(() => {...}, [])`

Code trong `useEffect` với `[]` chỉ chạy **1 lần duy nhất** sau khi component render lần đầu.

**Dùng khi**: Fetch initial data, setup subscriptions, initialize SDK.

```tsx
import React, { useState, useEffect } from "react";
import { View, Text, StyleSheet, ActivityIndicator } from "react-native";

interface UserData {
  name: string;
  email: string;
}

const UserProfile = () => {
  const [user, setUser] = useState<UserData | null>(null);
  const [loading, setLoading] = useState(true);

  // 🟢 MOUNT — Chạy 1 lần khi component xuất hiện
  useEffect(() => {
    console.log("🟢 Component MOUNTED");

    const fetchUser = async () => {
      try {
        const response = await fetch(
          "https://jsonplaceholder.typicode.com/users/1",
        );
        const data = await response.json();
        setUser({ name: data.name, email: data.email });
      } catch (err) {
        console.error("Failed to fetch user");
      } finally {
        setLoading(false);
      }
    };

    fetchUser();
  }, []); // 👈 Empty array = chỉ chạy khi mount

  if (loading) {
    return <ActivityIndicator size="large" style={styles.center} />;
  }

  return (
    <View style={styles.container}>
      <Text style={styles.name}>{user?.name}</Text>
      <Text style={styles.email}>{user?.email}</Text>
    </View>
  );
};

const styles = StyleSheet.create({
  center: { flex: 1, justifyContent: "center" },
  container: { padding: 16 },
  name: { fontSize: 20, fontWeight: "bold" },
  email: { color: "#666", marginTop: 4 },
});

export default UserProfile;
```

## 2. Update Phase — `useEffect(() => {...}, [dependencies])`

Chạy mỗi khi **dependency** trong mảng thay đổi.

**Dùng khi**: Fetch data mới khi params đổi, sync external system, log analytics.

```tsx
import React, { useState, useEffect } from "react";
import {
  View,
  Text,
  StyleSheet,
  TouchableOpacity,
  FlatList,
} from "react-native";

interface Post {
  id: number;
  title: string;
}

const PostsByUser = () => {
  const [userId, setUserId] = useState(1);
  const [posts, setPosts] = useState<Post[]>([]);
  const [loading, setLoading] = useState(false);

  // 🔄 UPDATE — Chạy mỗi khi userId thay đổi
  useEffect(() => {
    console.log(`🔄 Fetching posts for user ${userId}...`);
    let cancelled = false;

    const fetchPosts = async () => {
      setLoading(true);
      try {
        const response = await fetch(
          `https://jsonplaceholder.typicode.com/posts?userId=${userId}&_limit=5`,
        );
        const data = await response.json();
        if (!cancelled) {
          setPosts(data);
        }
      } finally {
        if (!cancelled) setLoading(false);
      }
    };

    fetchPosts();

    return () => {
      cancelled = true; // Cancel nếu userId đổi trước khi fetch xong
    };
  }, [userId]); // 👈 Dependency — chạy lại khi userId đổi

  return (
    <View style={styles.container}>
      <Text style={styles.title}>Posts by User {userId}</Text>

      {/* Nút chuyển user */}
      <View style={styles.row}>
        {[1, 2, 3, 4, 5].map((id) => (
          <TouchableOpacity
            key={id}
            style={[styles.userBtn, userId === id && styles.activeBtn]}
            onPress={() => setUserId(id)}
          >
            <Text
              style={[
                styles.userBtnText,
                userId === id && styles.activeBtnText,
              ]}
            >
              User {id}
            </Text>
          </TouchableOpacity>
        ))}
      </View>

      {/* Posts list */}
      {loading ? (
        <Text style={styles.loading}>Loading...</Text>
      ) : (
        <FlatList
          data={posts}
          keyExtractor={(item) => item.id.toString()}
          renderItem={({ item }) => (
            <View style={styles.postCard}>
              <Text style={styles.postTitle}>{item.title}</Text>
            </View>
          )}
        />
      )}
    </View>
  );
};

const styles = StyleSheet.create({
  container: { padding: 16 },
  title: { fontSize: 20, fontWeight: "bold", marginBottom: 12 },
  row: { flexDirection: "row", gap: 6, marginBottom: 16 },
  userBtn: {
    paddingHorizontal: 12,
    paddingVertical: 6,
    borderRadius: 16,
    backgroundColor: "#f0f0f0",
  },
  activeBtn: { backgroundColor: "#007AFF" },
  userBtnText: { fontSize: 12, color: "#333" },
  activeBtnText: { color: "#fff" },
  loading: { textAlign: "center", color: "#999", marginTop: 20 },
  postCard: {
    padding: 12,
    backgroundColor: "#fff",
    marginBottom: 8,
    borderRadius: 8,
  },
  postTitle: { fontSize: 14 },
});

export default PostsByUser;
```

## 3. Unmount Phase — Cleanup Function

Cleanup function chạy khi component bị **remove khỏi UI** hoặc **trước khi effect chạy lại**.

**Dùng khi**: Clear timers, remove event listeners, cancel subscriptions, cancel API requests.

```tsx
import React, { useState, useEffect } from "react";
import {
  View,
  Text,
  StyleSheet,
  TouchableOpacity,
  Dimensions,
} from "react-native";

// ========== Timer Component ==========
const Timer = () => {
  const [seconds, setSeconds] = useState(0);

  useEffect(() => {
    console.log("⏱ Timer started");

    const interval = setInterval(() => {
      setSeconds((prev) => prev + 1);
    }, 1000);

    // 🔴 UNMOUNT cleanup
    return () => {
      console.log("🔴 Timer cleaned up");
      clearInterval(interval);
    };
  }, []);

  return (
    <View style={styles.timerBox}>
      <Text style={styles.timerText}>⏱ {seconds}s</Text>
    </View>
  );
};

// ========== Window Resize Listener ==========
const WindowSize = () => {
  const [dimensions, setDimensions] = useState(Dimensions.get("window"));

  useEffect(() => {
    console.log("📐 Listener added");

    const subscription = Dimensions.addEventListener("change", ({ window }) => {
      setDimensions(window);
    });

    // 🔴 UNMOUNT cleanup
    return () => {
      console.log("📐 Listener removed");
      subscription.remove();
    };
  }, []);

  return (
    <View style={styles.dimBox}>
      <Text>
        📐 {Math.round(dimensions.width)} x {Math.round(dimensions.height)}
      </Text>
    </View>
  );
};

// ========== Parent — Toggle mount/unmount ==========
const LifecycleDemo = () => {
  const [showTimer, setShowTimer] = useState(true);
  const [showSize, setShowSize] = useState(true);

  return (
    <View style={styles.container}>
      <Text style={styles.title}>Component Lifecycle Demo</Text>
      <Text style={styles.subtitle}>Mở console để thấy mount/unmount logs</Text>

      {/* Toggle Timer */}
      <TouchableOpacity
        style={styles.toggleBtn}
        onPress={() => setShowTimer(!showTimer)}
      >
        <Text style={styles.toggleText}>
          {showTimer ? "🔴 Unmount Timer" : "🟢 Mount Timer"}
        </Text>
      </TouchableOpacity>
      {showTimer && <Timer />}

      {/* Toggle WindowSize */}
      <TouchableOpacity
        style={styles.toggleBtn}
        onPress={() => setShowSize(!showSize)}
      >
        <Text style={styles.toggleText}>
          {showSize ? "🔴 Unmount Size" : "🟢 Mount Size"}
        </Text>
      </TouchableOpacity>
      {showSize && <WindowSize />}
    </View>
  );
};

const styles = StyleSheet.create({
  container: { padding: 16, gap: 12 },
  title: { fontSize: 22, fontWeight: "bold" },
  subtitle: { color: "#666", marginBottom: 8 },
  toggleBtn: { backgroundColor: "#f0f0f0", padding: 12, borderRadius: 8 },
  toggleText: { fontSize: 16, textAlign: "center" },
  timerBox: { padding: 16, backgroundColor: "#e3f2fd", borderRadius: 8 },
  timerText: { fontSize: 20, fontWeight: "600", textAlign: "center" },
  dimBox: { padding: 16, backgroundColor: "#f3e5f5", borderRadius: 8 },
});

export default LifecycleDemo;
```

## 4. Conditional Effects

Thực thi logic khác nhau trong cùng 1 effect dựa trên điều kiện.

```tsx
import React, { useState, useEffect } from "react";
import { View, Text, Switch, StyleSheet } from "react-native";

const ConditionalEffectDemo = () => {
  const [isOnline, setIsOnline] = useState(true);
  const [data, setData] = useState<string | null>(null);

  useEffect(() => {
    if (isOnline) {
      console.log("🟢 Online — fetching fresh data...");
      // Fetch from API
      setData("Fresh data from API");
    } else {
      console.log("🔴 Offline — loading cached data...");
      // Load from cache
      setData("Cached data (offline)");
    }

    return () => {
      console.log("🧹 Cleaning up previous effect");
    };
  }, [isOnline]); // Chạy lại khi online status thay đổi

  return (
    <View style={styles.container}>
      <View style={styles.row}>
        <Text style={styles.label}>Online</Text>
        <Switch value={isOnline} onValueChange={setIsOnline} />
      </View>
      <Text style={styles.status}>
        Status: {isOnline ? "🟢 Online" : "🔴 Offline"}
      </Text>
      <Text style={styles.data}>Data: {data}</Text>
    </View>
  );
};

const styles = StyleSheet.create({
  container: { padding: 16, gap: 12 },
  row: {
    flexDirection: "row",
    justifyContent: "space-between",
    alignItems: "center",
  },
  label: { fontSize: 16, fontWeight: "600" },
  status: { fontSize: 16 },
  data: {
    fontSize: 14,
    color: "#666",
    backgroundColor: "#f9f9f9",
    padding: 12,
    borderRadius: 8,
  },
});
```

## 5. Sequence of Multiple useEffects

Nhiều `useEffect` trong 1 component — mỗi cái có mục đích riêng.

```tsx
import React, { useState, useEffect } from "react";
import { View, Text, StyleSheet, AppState } from "react-native";

const MultipleEffectsDemo = () => {
  const [user, setUser] = useState<string | null>(null);
  const [notifications, setNotifications] = useState(0);
  const [appState, setAppState] = useState(AppState.currentState);

  // Effect 1: Fetch user data — chỉ khi mount
  useEffect(() => {
    console.log("1️⃣ Fetching user...");
    // Simulate API call
    setTimeout(() => setUser("Hào"), 1000);
  }, []);

  // Effect 2: Fetch notifications — khi mount + mỗi khi user đổi
  useEffect(() => {
    if (!user) return;
    console.log(`2️⃣ Fetching notifications for ${user}...`);
    // Simulate API call
    setTimeout(() => setNotifications(5), 500);
  }, [user]);

  // Effect 3: Track app state — mount + cleanup
  useEffect(() => {
    console.log("3️⃣ Setting up app state listener");

    const subscription = AppState.addEventListener("change", (state) => {
      setAppState(state);
    });

    return () => {
      console.log("3️⃣ Removing app state listener");
      subscription.remove();
    };
  }, []);

  // Effect 4: Log analytics khi notifications thay đổi
  useEffect(() => {
    if (notifications > 0) {
      console.log(`4️⃣ Analytics: User has ${notifications} notifications`);
    }
  }, [notifications]);

  // Thứ tự chạy khi mount:
  // 1. Component render
  // 2. Effect 1 chạy (fetch user)
  // 3. Effect 2 chạy (skip vì user = null)
  // 4. Effect 3 chạy (setup listener)
  // 5. Effect 4 chạy (skip vì notifications = 0)
  // --- Sau 1s ---
  // 6. setUser('Hào') → re-render
  // 7. Effect 2 chạy lại (user đổi → fetch notifications)
  // --- Sau 0.5s ---
  // 8. setNotifications(5) → re-render
  // 9. Effect 4 chạy lại (notifications đổi → log analytics)

  return (
    <View style={styles.container}>
      <Text style={styles.title}>Multiple Effects Demo</Text>
      <Text>User: {user ?? "Loading..."}</Text>
      <Text>Notifications: {notifications}</Text>
      <Text>App State: {appState}</Text>
      <Text style={styles.hint}>Xem console để thấy thứ tự chạy</Text>
    </View>
  );
};

const styles = StyleSheet.create({
  container: { padding: 16, gap: 8 },
  title: { fontSize: 20, fontWeight: "bold", marginBottom: 8 },
  hint: { marginTop: 12, color: "#999", fontStyle: "italic", fontSize: 12 },
});

export default MultipleEffectsDemo;
```

**Quy tắc chia useEffect:**

- 🟢 **Tách riêng** — mỗi effect có 1 mục đích (SRP)
- 🟢 **Dependencies rõ ràng** — dễ hiểu khi nào chạy lại
- 🔴 **Không gộp** — nếu 2 effect có dependencies khác nhau

### Bài 1: Timer Component

**Yêu cầu**: Start timer on mount, cleanup on unmount.

```tsx
// exercises/TimerExercise.tsx
import React, { useState, useEffect } from "react";
import { View, Text, TouchableOpacity, StyleSheet } from "react-native";

// TODO: Tạo StopWatch component
// - Auto start khi mount
// - Hiển thị mm:ss format
// - Có nút Pause/Resume
// - Cleanup interval khi unmount

const StopWatch = () => {
  const [seconds, setSeconds] = useState(0);
  const [isRunning, setIsRunning] = useState(true);

  useEffect(() => {
    if (!isRunning) return;

    const interval = setInterval(() => {
      setSeconds((prev) => prev + 1);
    }, 1000);

    return () => clearInterval(interval);
  }, [isRunning]);

  const formatTime = (totalSeconds: number) => {
    const mins = Math.floor(totalSeconds / 60);
    const secs = totalSeconds % 60;
    return `${mins.toString().padStart(2, "0")}:${secs.toString().padStart(2, "0")}`;
  };

  return (
    <View style={styles.container}>
      <Text style={styles.time}>{formatTime(seconds)}</Text>
      <View style={styles.row}>
        <TouchableOpacity
          style={styles.btn}
          onPress={() => setIsRunning(!isRunning)}
        >
          <Text style={styles.btnText}>
            {isRunning ? "⏸ Pause" : "▶ Resume"}
          </Text>
        </TouchableOpacity>
        <TouchableOpacity
          style={[styles.btn, styles.resetBtn]}
          onPress={() => {
            setSeconds(0);
            setIsRunning(false);
          }}
        >
          <Text style={styles.btnText}>🔄 Reset</Text>
        </TouchableOpacity>
      </View>
    </View>
  );
};

const styles = StyleSheet.create({
  container: { padding: 20, alignItems: "center" },
  time: { fontSize: 48, fontWeight: "bold", fontVariant: ["tabular-nums"] },
  row: { flexDirection: "row", gap: 12, marginTop: 20 },
  btn: {
    backgroundColor: "#007AFF",
    paddingHorizontal: 24,
    paddingVertical: 12,
    borderRadius: 8,
  },
  resetBtn: { backgroundColor: "#6c757d" },
  btnText: { color: "#fff", fontWeight: "600", fontSize: 16 },
});

export default StopWatch;
```

### Bài 2: API Poll

**Yêu cầu**: Fetch data mỗi 30 giây, stop khi unmount.

```tsx
// exercises/APIPolling.tsx
import React, { useState, useEffect, useRef } from "react";
import { View, Text, StyleSheet, ActivityIndicator } from "react-native";

const APIPolling = () => {
  const [data, setData] = useState<string | null>(null);
  const [lastUpdated, setLastUpdated] = useState<Date | null>(null);
  const [pollCount, setPollCount] = useState(0);
  const isMounted = useRef(true);

  useEffect(() => {
    isMounted.current = true;

    const fetchData = async () => {
      try {
        console.log("📡 Polling...");
        const response = await fetch(
          "https://jsonplaceholder.typicode.com/posts/1",
        );
        const json = await response.json();

        if (isMounted.current) {
          setData(json.title);
          setLastUpdated(new Date());
          setPollCount((prev) => prev + 1);
        }
      } catch (err) {
        console.error("Poll failed:", err);
      }
    };

    // Fetch ngay lập tức
    fetchData();

    // Sau đó poll mỗi 30 giây
    const interval = setInterval(fetchData, 30000);

    // 🧹 Cleanup khi unmount
    return () => {
      isMounted.current = false;
      clearInterval(interval);
      console.log("🛑 Polling stopped");
    };
  }, []);

  return (
    <View style={styles.container}>
      <Text style={styles.title}>📡 API Polling</Text>
      <Text>Data: {data || "Loading..."}</Text>
      <Text>Poll count: {pollCount}</Text>
      <Text>
        Last updated: {lastUpdated?.toLocaleTimeString("vi-VN") || "Never"}
      </Text>
      <Text style={styles.hint}>Auto-refresh mỗi 30 giây</Text>
    </View>
  );
};

const styles = StyleSheet.create({
  container: { padding: 16, gap: 8 },
  title: { fontSize: 20, fontWeight: "bold", marginBottom: 8 },
  hint: { color: "#999", fontStyle: "italic", marginTop: 8 },
});

export default APIPolling;
```
