# 3.5 Navigation

> **Mục tiêu**: Sử dụng React Navigation để điều hướng giữa các màn hình trong React Native app.

---

## 📖 Tổng Quan

React Native **không có navigation built-in** — cần thư viện bên ngoài.  
**React Navigation** là thư viện phổ biến nhất.

### Các loại Navigator

| Navigator        | Mô tả                                  | Use case                    |
| ---------------- | -------------------------------------- | --------------------------- |
| **Stack**        | Push/pop screens (như browser history) | Flows: Home → Detail → Edit |
| **Tab**          | Bottom/Top tabs                        | Main app sections           |
| **Drawer**       | Side menu kéo ra                       | Dashboard, Settings         |
| **Native Stack** | Native animation (nhanh hơn)           | Production apps             |

## 2. Stack Navigator

### 2.1 Cơ Bản — Định Nghĩa Screens

```tsx
// navigation/RootNavigator.tsx
import React from "react";
import { createNativeStackNavigator } from "@react-navigation/native-stack";
import HomeScreen from "../screens/HomeScreen";
import DetailScreen from "../screens/DetailScreen";
import ProfileScreen from "../screens/ProfileScreen";

// ✅ Khai báo type cho params — TypeSafe navigation
export type RootStackParamList = {
  Home: undefined; // Không có params
  Detail: { id: number; title: string }; // Có params bắt buộc
  Profile: { userId: string } | undefined; // Params optional
};

const Stack = createNativeStackNavigator<RootStackParamList>();

export const RootNavigator = () => {
  return (
    <Stack.Navigator
      initialRouteName="Home"
      screenOptions={{
        headerStyle: { backgroundColor: "#007AFF" },
        headerTintColor: "#fff",
        headerTitleStyle: { fontWeight: "bold" },
        animation: "slide_from_right", // Animation khi chuyển màn hình
      }}
    >
      <Stack.Screen
        name="Home"
        component={HomeScreen}
        options={{ title: "Trang Chủ" }}
      />
      <Stack.Screen
        name="Detail"
        component={DetailScreen}
        options={({ route }) => ({
          title: route.params.title, // Dynamic title từ params
        })}
      />
      <Stack.Screen
        name="Profile"
        component={ProfileScreen}
        options={{
          headerShown: false, // Ẩn header
          presentation: "modal", // Hiển thị dạng modal
        }}
      />
    </Stack.Navigator>
  );
};
```

### 2.2 Navigate, GoBack, Push, Replace

```tsx
// screens/HomeScreen.tsx
import React from "react";
import {
  View,
  Text,
  TouchableOpacity,
  StyleSheet,
  FlatList,
} from "react-native";
import type { NativeStackScreenProps } from "@react-navigation/native-stack";
import type { RootStackParamList } from "../navigation/RootNavigator";

// ✅ Type-safe screen props
type Props = NativeStackScreenProps<RootStackParamList, "Home">;

const items = [
  { id: 1, title: "React Native Basics" },
  { id: 2, title: "Navigation Deep Dive" },
  { id: 3, title: "State Management" },
];

const HomeScreen = ({ navigation }: Props) => {
  return (
    <View style={styles.container}>
      <FlatList
        data={items}
        keyExtractor={(item) => item.id.toString()}
        renderItem={({ item }) => (
          <TouchableOpacity
            style={styles.item}
            onPress={() => {
              // ✅ navigate — chuyển đến screen (không tạo duplicate)
              navigation.navigate("Detail", {
                id: item.id,
                title: item.title,
              });
            }}
          >
            <Text style={styles.itemTitle}>{item.title}</Text>
            <Text style={styles.arrow}>→</Text>
          </TouchableOpacity>
        )}
        ItemSeparatorComponent={() => <View style={styles.separator} />}
      />

      <TouchableOpacity
        style={styles.profileBtn}
        onPress={() => {
          // ✅ navigate không cần params (optional)
          navigation.navigate("Profile");
        }}
      >
        <Text style={styles.profileBtnText}>👤 My Profile</Text>
      </TouchableOpacity>
    </View>
  );
};

const styles = StyleSheet.create({
  container: { flex: 1, backgroundColor: "#fff" },
  item: {
    flexDirection: "row",
    justifyContent: "space-between",
    alignItems: "center",
    padding: 16,
  },
  itemTitle: { fontSize: 16, fontWeight: "500" },
  arrow: { fontSize: 18, color: "#999" },
  separator: { height: 1, backgroundColor: "#f0f0f0", marginHorizontal: 16 },
  profileBtn: {
    padding: 16,
    backgroundColor: "#007AFF",
    margin: 16,
    borderRadius: 12,
    alignItems: "center",
  },
  profileBtnText: { color: "#fff", fontSize: 16, fontWeight: "600" },
});

export default HomeScreen;
```

### 2.3 Nhận Params & GoBack

```tsx
// screens/DetailScreen.tsx
import React from "react";
import { View, Text, TouchableOpacity, StyleSheet, Alert } from "react-native";
import type { NativeStackScreenProps } from "@react-navigation/native-stack";
import type { RootStackParamList } from "../navigation/RootNavigator";

type Props = NativeStackScreenProps<RootStackParamList, "Detail">;

const DetailScreen = ({ route, navigation }: Props) => {
  // ✅ Lấy params — type-safe!
  const { id, title } = route.params;

  return (
    <View style={styles.container}>
      <Text style={styles.id}>ID: {id}</Text>
      <Text style={styles.title}>{title}</Text>

      {/* Các cách điều hướng */}
      <View style={styles.buttons}>
        <TouchableOpacity
          style={styles.btn}
          onPress={() => navigation.goBack()} // ← Quay lại screen trước
        >
          <Text style={styles.btnText}>← Go Back</Text>
        </TouchableOpacity>

        <TouchableOpacity
          style={styles.btn}
          onPress={() => {
            // Push thêm 1 screen cùng loại lên stack
            navigation.push("Detail", { id: id + 1, title: `Item ${id + 1}` });
          }}
        >
          <Text style={styles.btnText}>Push Next →</Text>
        </TouchableOpacity>

        <TouchableOpacity
          style={[styles.btn, { backgroundColor: "#ff6b6b" }]}
          onPress={() => {
            // Replace — thay thế screen hiện tại (không thể goBack)
            navigation.replace("Detail", { id: 999, title: "Replaced!" });
          }}
        >
          <Text style={styles.btnText}>Replace</Text>
        </TouchableOpacity>

        <TouchableOpacity
          style={[styles.btn, { backgroundColor: "#4ecdc4" }]}
          onPress={() => {
            // popToTop — về screen đầu tiên của stack
            navigation.popToTop();
          }}
        >
          <Text style={styles.btnText}>Pop to Home</Text>
        </TouchableOpacity>
      </View>

      {/* Set custom header options dynamically */}
      <TouchableOpacity
        onPress={() => {
          navigation.setOptions({
            title: "Updated Title!",
            headerRight: () => (
              <TouchableOpacity onPress={() => Alert.alert("Menu pressed")}>
                <Text style={{ color: "#fff", fontSize: 18 }}>⋯</Text>
              </TouchableOpacity>
            ),
          });
        }}
      >
        <Text style={{ color: "#007AFF", marginTop: 20 }}>
          Tap to update header
        </Text>
      </TouchableOpacity>
    </View>
  );
};

const styles = StyleSheet.create({
  container: { flex: 1, padding: 20, backgroundColor: "#fff" },
  id: { fontSize: 14, color: "#999" },
  title: { fontSize: 24, fontWeight: "bold", marginTop: 4, marginBottom: 20 },
  buttons: { gap: 12 },
  btn: {
    backgroundColor: "#007AFF",
    padding: 14,
    borderRadius: 10,
    alignItems: "center",
  },
  btnText: { color: "#fff", fontSize: 16, fontWeight: "600" },
});

export default DetailScreen;
```

## 4. Nested Navigation (Stack + Tab)

Pattern phổ biến: **Tabs ở ngoài → Stack bên trong mỗi Tab**.

```tsx
// navigation/AppNavigator.tsx
import React from "react";
import { createNativeStackNavigator } from "@react-navigation/native-stack";
import { createBottomTabNavigator } from "@react-navigation/bottom-tabs";
import { Text } from "react-native";

// Screens (simplified)
import HomeScreen from "../screens/HomeScreen";
import DetailScreen from "../screens/DetailScreen";
import ProfileScreen from "../screens/ProfileScreen";

// ===== Types =====
export type HomeStackParamList = {
  HomeList: undefined;
  Detail: { id: number; title: string };
};

export type ProfileStackParamList = {
  ProfileMain: undefined;
  EditProfile: undefined;
};

export type RootTabParamList = {
  HomeStack: undefined;
  ProfileStack: undefined;
};

// ===== Home Stack =====
const HomeStack = createNativeStackNavigator<HomeStackParamList>();

const HomeStackNavigator = () => (
  <HomeStack.Navigator>
    <HomeStack.Screen
      name="HomeList"
      component={HomeScreen}
      options={{ title: "Home" }}
    />
    <HomeStack.Screen name="Detail" component={DetailScreen} />
  </HomeStack.Navigator>
);

// ===== Profile Stack =====
const ProfileStack = createNativeStackNavigator<ProfileStackParamList>();

const ProfileStackNavigator = () => (
  <ProfileStack.Navigator>
    <ProfileStack.Screen
      name="ProfileMain"
      component={ProfileScreen}
      options={{ title: "Profile" }}
    />
    <ProfileStack.Screen
      name="EditProfile"
      component={ProfileScreen}
      options={{ title: "Edit Profile" }}
    />
  </ProfileStack.Navigator>
);

// ===== Root Tab =====
const RootTab = createBottomTabNavigator<RootTabParamList>();

export const AppNavigator = () => (
  <RootTab.Navigator screenOptions={{ headerShown: false }}>
    <RootTab.Screen
      name="HomeStack"
      component={HomeStackNavigator}
      options={{
        title: "Home",
        tabBarIcon: ({ color, size }) => (
          <Text style={{ fontSize: size, color }}>🏠</Text>
        ),
      }}
    />
    <RootTab.Screen
      name="ProfileStack"
      component={ProfileStackNavigator}
      options={{
        title: "Profile",
        tabBarIcon: ({ color, size }) => (
          <Text style={{ fontSize: size, color }}>👤</Text>
        ),
      }}
    />
  </RootTab.Navigator>
);
```

## 6. useNavigation & useRoute Hooks

Dùng khi component **không phải screen** (không có `navigation` prop).

```tsx
import React from "react";
import { TouchableOpacity, Text } from "react-native";
import { useNavigation, useRoute } from "@react-navigation/native";
import type { NativeStackNavigationProp } from "@react-navigation/native-stack";
import type { RouteProp } from "@react-navigation/native";
import type { RootStackParamList } from "../navigation/RootNavigator";

// ===== useNavigation =====
type NavigationProp = NativeStackNavigationProp<RootStackParamList>;

const GoToDetailButton = ({ id, title }: { id: number; title: string }) => {
  const navigation = useNavigation<NavigationProp>();

  return (
    <TouchableOpacity
      onPress={() => navigation.navigate("Detail", { id, title })}
      style={{ padding: 12, backgroundColor: "#007AFF", borderRadius: 8 }}
    >
      <Text style={{ color: "#fff" }}>Go to {title}</Text>
    </TouchableOpacity>
  );
};

// ===== useRoute =====
type DetailRouteProp = RouteProp<RootStackParamList, "Detail">;

const DetailInfo = () => {
  const route = useRoute<DetailRouteProp>();
  const { id, title } = route.params;

  return (
    <Text>
      Detail: {id} — {title}
    </Text>
  );
};
```

## 7. Authentication Flow

Pattern phổ biến: **2 stack khác nhau** cho Logged In / Logged Out.

```tsx
// navigation/AuthNavigator.tsx
import React from "react";
import { createNativeStackNavigator } from "@react-navigation/native-stack";
import { View, Text, TouchableOpacity, StyleSheet } from "react-native";

// ===== Types =====
type AuthStackParamList = {
  Login: undefined;
  Register: undefined;
};

type AppStackParamList = {
  Dashboard: undefined;
  Settings: undefined;
};

// ===== Auth Screens =====
const LoginScreen = ({ onLogin }: { onLogin: () => void }) => (
  <View style={authStyles.container}>
    <Text style={authStyles.title}>Login</Text>
    <TouchableOpacity style={authStyles.btn} onPress={onLogin}>
      <Text style={authStyles.btnText}>Sign In</Text>
    </TouchableOpacity>
  </View>
);

const DashboardScreen = ({ onLogout }: { onLogout: () => void }) => (
  <View style={authStyles.container}>
    <Text style={authStyles.title}>Dashboard ✅</Text>
    <TouchableOpacity
      style={[authStyles.btn, { backgroundColor: "#ff3b30" }]}
      onPress={onLogout}
    >
      <Text style={authStyles.btnText}>Sign Out</Text>
    </TouchableOpacity>
  </View>
);

const authStyles = StyleSheet.create({
  container: {
    flex: 1,
    justifyContent: "center",
    alignItems: "center",
    gap: 20,
  },
  title: { fontSize: 28, fontWeight: "bold" },
  btn: {
    backgroundColor: "#007AFF",
    paddingHorizontal: 32,
    paddingVertical: 14,
    borderRadius: 12,
  },
  btnText: { color: "#fff", fontSize: 16, fontWeight: "600" },
});

// ===== Navigators =====
const AuthStack = createNativeStackNavigator<AuthStackParamList>();
const AppStack = createNativeStackNavigator<AppStackParamList>();

// ===== Root — conditional rendering =====
import { useState } from "react";

export const RootAuthNavigator = () => {
  const [isLoggedIn, setIsLoggedIn] = useState(false);

  // React Navigation tự xử lý animation khi switch stack
  if (!isLoggedIn) {
    return (
      <AuthStack.Navigator screenOptions={{ headerShown: false }}>
        <AuthStack.Screen name="Login">
          {() => <LoginScreen onLogin={() => setIsLoggedIn(true)} />}
        </AuthStack.Screen>
      </AuthStack.Navigator>
    );
  }

  return (
    <AppStack.Navigator>
      <AppStack.Screen name="Dashboard">
        {() => <DashboardScreen onLogout={() => setIsLoggedIn(false)} />}
      </AppStack.Screen>
    </AppStack.Navigator>
  );
};
```

## 8. Deep Linking (Cơ Bản)

```tsx
// App.tsx
import { NavigationContainer } from "@react-navigation/native";

const linking = {
  prefixes: ["myapp://", "https://myapp.com"],
  config: {
    screens: {
      Home: "home",
      Detail: "detail/:id", // myapp://detail/42
      Profile: "profile/:userId", // myapp://profile/abc123
    },
  },
};

export default function App() {
  return (
    <NavigationContainer linking={linking}>
      {/* Navigators */}
    </NavigationContainer>
  );
}

// Test: npx uri-scheme open myapp://detail/42 --android
```
