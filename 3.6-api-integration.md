# 3.6 API Integration

> **Mục tiêu**: Gọi API, xử lý dữ liệu, error handling và tạo custom hooks cho data fetching trong React Native.

---

## 📖 Tổng Quan

React Native sử dụng **Fetch API** (built-in) hoặc **Axios** (thư viện) để gọi HTTP requests.

|                 | Fetch API                   | Axios                         |
| --------------- | --------------------------- | ----------------------------- |
| Built-in        | ✅                          | ❌ (cài thêm)                 |
| JSON auto-parse | ❌ (phải `.json()`)         | ✅                            |
| Interceptors    | ❌                          | ✅                            |
| Cancel request  | AbortController             | CancelToken / AbortController |
| Error handling  | Chỉ reject khi network fail | Reject khi status >= 400      |
| Timeout         | Không built-in              | ✅                            |

```tsx
  View,
  Text,
  FlatList,
  StyleSheet,
  ActivityIndicator,
  Image,
} from "react-native";

interface User {
  id: number;
  name: string;
  email: string;
  avatar: string;
}

// Dùng JSONPlaceholder hoặc ReqRes API
const API_URL = "https://reqres.in/api/users";

const FetchGetDemo = () => {
  const [users, setUsers] = useState<User[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    fetchUsers();
  }, []);

  const fetchUsers = async () => {
    try {
      setLoading(true);
      setError(null);

      const response = await fetch(`${API_URL}?page=1`);

      // ⚠️ Fetch KHÔNG throw error cho 4xx/5xx — phải check manually
      if (!response.ok) {
        throw new Error(`HTTP ${response.status}: ${response.statusText}`);
      }

      const json = await response.json();
      const mappedUsers: User[] = json.data.map((u: any) => ({
        id: u.id,
        name: `${u.first_name} ${u.last_name}`,

      }));

      setUsers(mappedUsers);
    } catch (err) {
      setError(err instanceof Error ? err.message : "Unknown error");
    } finally {
      setLoading(false);
    }
  };

  if (loading) {
    return (
      <View style={styles.center}>
        <ActivityIndicator size="large" color="#007AFF" />
        <Text style={{ marginTop: 8 }}>Loading...</Text>
      </View>
    );
  }

  if (error) {
    return (
      <View style={styles.center}>
        <Text style={styles.errorText}>❌ {error}</Text>
      </View>
    );
  }

  return (
    <FlatList
      data={users}
      keyExtractor={(item) => item.id.toString()}
      renderItem={({ item }) => (
        <View style={styles.userItem}>
          <Image source={{ uri: item.avatar }} style={styles.avatar} />

      ItemSeparatorComponent={() => <View style={styles.separator} />}
    />
  );
};

  center: { flex: 1, justifyContent: "center", alignItems: "center" },

    flexDirection: "row",
    alignItems: "center",
    padding: 16,
```

### 1.2 POST, PUT, DELETE

```tsx
// API service functions
const API_BASE = "https://jsonplaceholder.typicode.com";

interface Post {
  id?: number;
  title: string;
  body: string;
  userId: number;
}

// POST — Tạo mới
const createPost = async (post: Omit<Post, "id">): Promise<Post> => {
  const response = await fetch(`${API_BASE}/posts`, {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
      // Authorization: `Bearer ${token}`,  // Nếu cần auth
    },
    body: JSON.stringify(post),
  });

  if (!response.ok) throw new Error(`Create failed: ${response.status}`);
  return response.json();
};

// PUT — Cập nhật toàn bộ
const updatePost = async (id: number, post: Post): Promise<Post> => {
  const response = await fetch(`${API_BASE}/posts/${id}`, {
    method: "PUT",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(post),
  });

  if (!response.ok) throw new Error(`Update failed: ${response.status}`);
  return response.json();
};

// PATCH — Cập nhật 1 phần
const patchPost = async (id: number, partial: Partial<Post>): Promise<Post> => {
  const response = await fetch(`${API_BASE}/posts/${id}`, {
    method: "PATCH",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(partial),
  });

  if (!response.ok) throw new Error(`Patch failed: ${response.status}`);
  return response.json();
};

// DELETE
const deletePost = async (id: number): Promise<void> => {
  const response = await fetch(`${API_BASE}/posts/${id}`, {
    method: "DELETE",
  });

  if (!response.ok) throw new Error(`Delete failed: ${response.status}`);
};
```

### 1.3 Cancel Request (AbortController)

```tsx
import React, { useState, useEffect } from "react";
import { View, Text, TextInput, FlatList, StyleSheet } from "react-native";

const SearchWithCancel = () => {
  const [query, setQuery] = useState("");
  const [results, setResults] = useState<any[]>([]);

  useEffect(() => {
    if (!query.trim()) {
      setResults([]);
      return;
    }

    // ✅ AbortController — cancel request khi query thay đổi
    const controller = new AbortController();

    const search = async () => {
      try {
        const res = await fetch(
          `https://jsonplaceholder.typicode.com/posts?title_like=${query}`,
          { signal: controller.signal }, // Gắn signal
        );
        const data = await res.json();
        setResults(data.slice(0, 10));
      } catch (err: any) {
        if (err.name === "AbortError") {
          console.log("Request cancelled"); // Bình thường, không phải error
          return;
        }
        console.error(err);
      }
    };

    // Debounce 300ms
    const timer = setTimeout(search, 300);

    return () => {
      clearTimeout(timer);
      controller.abort(); // Cancel request cũ khi cleanup
    };
  }, [query]);

  return (
    <View style={{ flex: 1, padding: 16 }}>
      <TextInput
        style={searchStyles.input}
        placeholder="Search posts..."
        value={query}
        onChangeText={setQuery}
      />
      <FlatList
        data={results}
        keyExtractor={(item) => item.id.toString()}
        renderItem={({ item }) => (
          <Text style={searchStyles.item}>{item.title}</Text>
        )}
      />
    </View>
  );
};

const searchStyles = StyleSheet.create({
  input: {
    borderWidth: 1,
    borderColor: "#ddd",
    borderRadius: 8,
    padding: 12,
    marginBottom: 12,
    fontSize: 16,
  },
  item: {
    paddingVertical: 12,
    borderBottomWidth: 1,
    borderBottomColor: "#f0f0f0",
  },
});
```

### 2.1 Cài Đặt & Instance

```tsx
// services/api.ts
import axios from "axios";

// ✅ Tạo instance — config 1 lần, dùng nhiều nơi
const api = axios.create({
  baseURL: "https://reqres.in/api",
  timeout: 10000, // 10 giây
  headers: {
    "Content-Type": "application/json",
  },
});

// ===== Interceptors =====

// Request interceptor — thêm token vào mỗi request
api.interceptors.request.use(
  async (config) => {
    // Lấy token từ storage
    // const token = await AsyncStorage.getItem('token');
    const token = "your-auth-token";
    if (token) {
      config.headers.Authorization = `Bearer ${token}`;
    }
    console.log(`[API] ${config.method?.toUpperCase()} ${config.url}`);
    return config;
  },
  (error) => Promise.reject(error),
);

// Response interceptor — xử lý lỗi chung
api.interceptors.response.use(
  (response) => response,
  async (error) => {
    if (error.response) {
      // Server responded với status code lỗi
      switch (error.response.status) {
        case 401:
          console.log("Unauthorized — redirect to login");
          // Navigation.navigate('Login');
          break;
        case 403:
          console.log("Forbidden");
          break;
        case 404:
          console.log("Not found");
          break;
        case 500:
          console.log("Server error");
          break;
      }
    } else if (error.request) {
      // Request đã gửi nhưng không nhận response (network error)
      console.log("Network error — no response");
    }
    return Promise.reject(error);
  },
);

export default api;
```

## 3. Data Fetching Patterns

### 3.1 Loading / Error / Empty States

```tsx
import React, { useState, useEffect, useCallback } from "react";
import {
  View,
  Text,
  FlatList,
  StyleSheet,
  ActivityIndicator,
  TouchableOpacity,
  RefreshControl,
} from "react-native";
import { getUsers, User } from "../services/userService";

interface FetchState<T> {
  data: T[];
  loading: boolean;
  error: string | null;
  refreshing: boolean;
}

const UserListScreen = () => {
  const [state, setState] = useState<FetchState<User>>({
    data: [],
    loading: true,
    error: null,
    refreshing: false,
  });

  const fetchData = useCallback(async (isRefresh = false) => {
    try {
      setState((prev) => ({
        ...prev,
        loading: !isRefresh,
        refreshing: isRefresh,
        error: null,
      }));

      const response = await getUsers(1);
      setState((prev) => ({
        ...prev,
        data: response.data,
        loading: false,
        refreshing: false,
      }));
    } catch (err) {
      setState((prev) => ({
        ...prev,
        error: err instanceof Error ? err.message : "Something went wrong",
        loading: false,
        refreshing: false,
      }));
    }
  }, []);

  useEffect(() => {
    fetchData();
  }, [fetchData]);

  // ===== LOADING STATE =====
  if (state.loading) {
    return (
      <View style={pStyles.center}>
        <ActivityIndicator size="large" color="#007AFF" />
        <Text style={pStyles.loadingText}>Loading users...</Text>
      </View>
    );
  }

  // ===== ERROR STATE =====
  if (state.error) {
    return (
      <View style={pStyles.center}>
        <Text style={pStyles.errorIcon}>⚠️</Text>
        <Text style={pStyles.errorText}>{state.error}</Text>
        <TouchableOpacity style={pStyles.retryBtn} onPress={() => fetchData()}>
          <Text style={pStyles.retryBtnText}>Retry</Text>
        </TouchableOpacity>
      </View>
    );
  }

  // ===== EMPTY STATE =====
  if (state.data.length === 0) {
    return (
      <View style={pStyles.center}>
        <Text style={pStyles.emptyIcon}>📭</Text>
        <Text style={pStyles.emptyText}>No users found</Text>
      </View>
    );
  }

  // ===== DATA STATE =====
  return (
    <FlatList
      data={state.data}
      keyExtractor={(item) => item.id.toString()}
      renderItem={({ item }) => (
        <View style={pStyles.item}>
          <Text style={pStyles.name}>
            {item.first_name} {item.last_name}
          </Text>
          <Text style={pStyles.email}>{item.email}</Text>
        </View>
      )}
      refreshControl={
        <RefreshControl
          refreshing={state.refreshing}
          onRefresh={() => fetchData(true)}
          colors={["#007AFF"]}
        />
      }
    />
  );
};

const pStyles = StyleSheet.create({
  center: {
    flex: 1,
    justifyContent: "center",
    alignItems: "center",
    padding: 20,
  },
  loadingText: { marginTop: 12, color: "#666" },
  errorIcon: { fontSize: 48, marginBottom: 12 },
  errorText: {
    fontSize: 16,
    color: "#dc3545",
    textAlign: "center",
    marginBottom: 16,
  },
  retryBtn: {
    backgroundColor: "#007AFF",
    paddingHorizontal: 24,
    paddingVertical: 12,
    borderRadius: 8,
  },
  retryBtnText: { color: "#fff", fontWeight: "600" },
  emptyIcon: { fontSize: 48, marginBottom: 12 },
  emptyText: { fontSize: 16, color: "#999" },
  item: { padding: 16, borderBottomWidth: 1, borderBottomColor: "#f0f0f0" },
  name: { fontSize: 16, fontWeight: "600" },
  email: { fontSize: 14, color: "#666", marginTop: 2 },
});
```

### 3.2 Infinite Scroll (Pagination)

```tsx
import React, { useState, useEffect, useCallback } from "react";
import {
  View,
  Text,
  FlatList,
  ActivityIndicator,
  StyleSheet,
} from "react-native";

interface Post {
  id: number;
  title: string;
  body: string;
}

const PAGE_SIZE = 10;

const InfiniteScrollDemo = () => {
  const [posts, setPosts] = useState<Post[]>([]);
  const [page, setPage] = useState(1);
  const [loading, setLoading] = useState(false);
  const [hasMore, setHasMore] = useState(true);

  const fetchPosts = useCallback(
    async (pageNum: number) => {
      if (loading || !hasMore) return;

      setLoading(true);
      try {
        const res = await fetch(
          `https://jsonplaceholder.typicode.com/posts?_page=${pageNum}&_limit=${PAGE_SIZE}`,
        );
        const data: Post[] = await res.json();

        if (data.length < PAGE_SIZE) {
          setHasMore(false); // Hết data
        }

        setPosts((prev) => [...prev, ...data]);
      } catch (err) {
        console.error(err);
      } finally {
        setLoading(false);
      }
    },
    [loading, hasMore],
  );

  useEffect(() => {
    fetchPosts(1);
  }, []);

  const loadMore = () => {
    if (!loading && hasMore) {
      const nextPage = page + 1;
      setPage(nextPage);
      fetchPosts(nextPage);
    }
  };

  const renderFooter = () => {
    if (!loading) return null;
    return (
      <View style={{ padding: 16, alignItems: "center" }}>
        <ActivityIndicator size="small" color="#007AFF" />
      </View>
    );
  };

  return (
    <FlatList
      data={posts}
      keyExtractor={(item) => item.id.toString()}
      renderItem={({ item }) => (
        <View style={isStyles.item}>
          <Text style={isStyles.title}>{item.title}</Text>
          <Text style={isStyles.body} numberOfLines={2}>
            {item.body}
          </Text>
        </View>
      )}
      onEndReached={loadMore} // Trigger khi scroll gần cuối
      onEndReachedThreshold={0.3} // Trigger khi còn 30% list
      ListFooterComponent={renderFooter}
      ItemSeparatorComponent={() => <View style={isStyles.sep} />}
    />
  );
};

const isStyles = StyleSheet.create({
  item: { padding: 16 },
  title: { fontSize: 15, fontWeight: "600", marginBottom: 4 },
  body: { fontSize: 13, color: "#666", lineHeight: 18 },
  sep: { height: 1, backgroundColor: "#f0f0f0" },
});
```

## 4. Custom Hook — useFetch

Tái sử dụng logic fetch ở mọi nơi.

```tsx
// hooks/useFetch.ts
import { useState, useEffect, useCallback } from "react";

interface UseFetchResult<T> {
  data: T | null;
  loading: boolean;
  error: string | null;
  refetch: () => Promise<void>;
}

export function useFetch<T>(
  url: string,
  options?: RequestInit,
): UseFetchResult<T> {
  const [data, setData] = useState<T | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const fetchData = useCallback(async () => {
    const controller = new AbortController();

    try {
      setLoading(true);
      setError(null);

      const response = await fetch(url, {
        ...options,
        signal: controller.signal,
      });

      if (!response.ok) {
        throw new Error(`HTTP ${response.status}`);
      }

      const json = await response.json();
      setData(json);
    } catch (err: any) {
      if (err.name !== "AbortError") {
        setError(err.message || "Unknown error");
      }
    } finally {
      setLoading(false);
    }

    return () => controller.abort();
  }, [url]);

  useEffect(() => {
    fetchData();
  }, [fetchData]);

  return { data, loading, error, refetch: fetchData };
}

// ===== Sử dụng =====
```

### Custom Hook — useApi (nâng cao)

```tsx
// hooks/useApi.ts
import { useState, useCallback } from "react";

interface UseApiResult<T> {
  data: T | null;
  loading: boolean;
  error: string | null;
  execute: (...args: any[]) => Promise<T | undefined>;
  reset: () => void;
}

export function useApi<T>(
  apiFunction: (...args: any[]) => Promise<T>,
): UseApiResult<T> {
  const [data, setData] = useState<T | null>(null);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const execute = useCallback(
    async (...args: any[]): Promise<T | undefined> => {
      try {
        setLoading(true);
        setError(null);
        const result = await apiFunction(...args);
        setData(result);
        return result;
      } catch (err: any) {
        const message = err.response?.data?.message || err.message || "Error";
        setError(message);
        return undefined;
      } finally {
        setLoading(false);
      }
    },
    [apiFunction],
  );

  const reset = useCallback(() => {
    setData(null);
    setLoading(false);
    setError(null);
  }, []);

  return { data, loading, error, execute, reset };
}

// ===== Sử dụng =====
import { createUser } from "../services/userService";

const CreateUserScreen = () => {
  const { data, loading, error, execute } = useApi(createUser);

  const handleSubmit = async () => {
    const result = await execute({ name: "John", job: "Developer" });
    if (result) {
      console.log("Created:", result);
    }
  };

  return (
    <View>
      {loading && <ActivityIndicator />}
      {error && <Text style={{ color: "red" }}>{error}</Text>}
      {data && <Text>Created successfully! ID: {data.id}</Text>}
      <TouchableOpacity onPress={handleSubmit} disabled={loading}>
        <Text>Create User</Text>
      </TouchableOpacity>
    </View>
  );
};
```

## 5. Error Handling Best Practices

```tsx
// utils/errorHandler.ts

export class ApiError extends Error {
  status: number;
  data: any;

  constructor(status: number, message: string, data?: any) {
    super(message);
    this.name = "ApiError";
    this.status = status;
    this.data = data;
  }
}

// Centralized error handler
export const handleApiError = (error: unknown): string => {
  if (error instanceof ApiError) {
    switch (error.status) {
      case 400:
        return "Dữ liệu không hợp lệ. Vui lòng kiểm tra lại.";
      case 401:
        return "Phiên đăng nhập đã hết hạn. Vui lòng đăng nhập lại.";
      case 403:
        return "Bạn không có quyền thực hiện thao tác này.";
      case 404:
        return "Không tìm thấy dữ liệu.";
      case 429:
        return "Quá nhiều yêu cầu. Vui lòng thử lại sau.";
      case 500:
        return "Lỗi máy chủ. Vui lòng thử lại sau.";
      default:
        return `Lỗi: ${error.message}`;
    }
  }

  if (
    error instanceof TypeError &&
    error.message === "Network request failed"
  ) {
    return "Không có kết nối mạng. Vui lòng kiểm tra và thử lại.";
  }

  return "Đã xảy ra lỗi. Vui lòng thử lại.";
};
```
