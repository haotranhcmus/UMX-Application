# 3.4 Flexbox Layout

> **Mục tiêu**: Master Flexbox layout trong React Native — sắp xếp UI chính xác cho mọi màn hình.

---

## 📖 Tổng Quan

React Native dùng **Flexbox** (giống CSS Flexbox) nhưng có khác biệt quan trọng:

|                         | CSS Flexbox         | React Native Flexbox                |
| ----------------------- | ------------------- | ----------------------------------- |
| Default `flexDirection` | `row`               | **`column`**                        |
| Default `alignContent`  | `stretch`           | **`flex-start`**                    |
| `flex`                  | Shorthand 3 giá trị | **Chỉ 1 số**                        |
| Đơn vị                  | px, %, rem...       | **dp (density-independent pixels)** |

---

## 1. Container Properties (Parent)

### 1.1 flexDirection

Xác định hướng **main axis** (trục chính).

```tsx
import React, { useState } from "react";
import { View, Text, TouchableOpacity, StyleSheet } from "react-native";

type Direction = "column" | "row" | "column-reverse" | "row-reverse";

const FlexDirectionDemo = () => {
  const [dir, setDir] = useState<Direction>("column");
  const directions: Direction[] = [
    "column",
    "row",
    "column-reverse",
    "row-reverse",
  ];

  return (
    <View style={styles.container}>
      <View style={styles.buttons}>
        {directions.map((d) => (
          <TouchableOpacity
            key={d}
            style={[styles.btn, dir === d && styles.btnActive]}
            onPress={() => setDir(d)}
          >
            <Text style={[styles.btnText, dir === d && { color: "#fff" }]}>
              {d}
            </Text>
          </TouchableOpacity>
        ))}
      </View>

      {/* 👇 Main axis thay đổi theo flexDirection */}
      <View style={[styles.box, { flexDirection: dir }]}>
        <View style={[styles.item, { backgroundColor: "#ff6b6b" }]}>
          <Text style={styles.itemText}>1</Text>
        </View>
        <View style={[styles.item, { backgroundColor: "#4ecdc4" }]}>
          <Text style={styles.itemText}>2</Text>
        </View>
        <View style={[styles.item, { backgroundColor: "#45b7d1" }]}>
          <Text style={styles.itemText}>3</Text>
        </View>
      </View>
    </View>
  );
};

const styles = StyleSheet.create({
  container: { padding: 16, gap: 12 },
  buttons: { flexDirection: "row", flexWrap: "wrap", gap: 8 },
  btn: {
    paddingHorizontal: 12,
    paddingVertical: 6,
    borderRadius: 8,
    borderWidth: 1,
    borderColor: "#007AFF",
  },
  btnActive: { backgroundColor: "#007AFF" },
  btnText: { fontSize: 12, color: "#007AFF" },
  box: {
    height: 200,
    backgroundColor: "#f0f0f0",
    borderRadius: 8,
    padding: 8,
    gap: 8,
  },
  item: {
    width: 50,
    height: 50,
    borderRadius: 8,
    justifyContent: "center",
    alignItems: "center",
  },
  itemText: { color: "#fff", fontWeight: "bold" },
});
```

### 1.2 justifyContent (Main Axis)

Canh chỉnh items **dọc theo main axis**.

```
flexDirection: 'column' → Main axis = ↓ (vertical)
flexDirection: 'row'    → Main axis = → (horizontal)
```

```tsx
import React from "react";
import { View, Text, StyleSheet, ScrollView } from "react-native";

type JustifyValue =
  | "flex-start"
  | "flex-end"
  | "center"
  | "space-between"
  | "space-around"
  | "space-evenly";

const JustifyContentDemo = () => {
  const values: JustifyValue[] = [
    "flex-start",
    "flex-end",
    "center",
    "space-between",
    "space-around",
    "space-evenly",
  ];

  return (
    <ScrollView style={{ padding: 16 }}>
      {values.map((jc) => (
        <View key={jc} style={{ marginBottom: 16 }}>
          <Text style={styles.label}>{jc}</Text>
          <View style={[styles.row, { justifyContent: jc }]}>
            {[1, 2, 3].map((i) => (
              <View key={i} style={styles.item}>
                <Text style={styles.itemText}>{i}</Text>
              </View>
            ))}
          </View>
        </View>
      ))}
    </ScrollView>
  );
};

const styles = StyleSheet.create({
  label: { fontSize: 13, fontWeight: "600", color: "#666", marginBottom: 4 },
  row: {
    flexDirection: "row",
    backgroundColor: "#f0f0f0",
    height: 60,
    borderRadius: 8,
    padding: 4,
    alignItems: "center",
  },
  item: {
    width: 40,
    height: 40,
    backgroundColor: "#007AFF",
    borderRadius: 6,
    justifyContent: "center",
    alignItems: "center",
  },
  itemText: { color: "#fff", fontWeight: "bold", fontSize: 12 },
});
```

### 1.3 alignItems (Cross Axis)

Canh chỉnh items **vuông góc với main axis**.

```
flexDirection: 'column' → Cross axis = → (horizontal)
flexDirection: 'row'    → Cross axis = ↓ (vertical)
```

```tsx
import React from "react";
import { View, Text, StyleSheet, ScrollView } from "react-native";

const AlignItemsDemo = () => {
  const values = [
    "flex-start",
    "flex-end",
    "center",
    "stretch",
    "baseline",
  ] as const;

  return (
    <ScrollView style={{ padding: 16 }}>
      {values.map((ai) => (
        <View key={ai} style={{ marginBottom: 16 }}>
          <Text style={styles.label}>alignItems: '{ai}'</Text>
          <View style={[styles.box, { alignItems: ai }]}>
            <View style={[styles.item, { height: 30 }]}>
              <Text style={styles.t}>A</Text>
            </View>
            <View style={[styles.item, { height: 50 }]}>
              <Text style={styles.t}>B</Text>
            </View>
            <View style={[styles.item, { height: 40 }]}>
              <Text style={styles.t}>C</Text>
            </View>
          </View>
        </View>
      ))}
    </ScrollView>
  );
};

const styles = StyleSheet.create({
  label: { fontSize: 13, fontWeight: "600", color: "#666", marginBottom: 4 },
  box: {
    flexDirection: "row",
    backgroundColor: "#f0f0f0",
    height: 80,

});
```

### 1.4 flexWrap & alignContent


## 3. Common Layout Patterns

### 3.1 Center (Absolute Center)

```tsx
const CenterLayout = () => (
  <View
    style={{
      flex: 1,
      justifyContent: "center", // Main axis → vertical center
      alignItems: "center", // Cross axis → horizontal center
      backgroundColor: "#f0f0f0",
    }}
  >
    <View
      style={{
        width: 200,
        height: 100,
        backgroundColor: "#007AFF",
        borderRadius: 12,
        justifyContent: "center",
        alignItems: "center",
      }}
    >
      <Text style={{ color: "#fff", fontWeight: "bold" }}>Centered!</Text>
    </View>
  </View>
);
```

### 3.2 Header — Content — Footer

```tsx
import React from "react";
import { View, Text, StyleSheet, ScrollView } from "react-native";

const HeaderContentFooter = () => (
  <View style={{ flex: 1 }}>
    {/* Header — kích thước cố định */}
    <View style={lcStyles.header}>
      <Text style={lcStyles.headerText}>Header</Text>
    </View>

    {/* Content — flex: 1 → chiếm toàn bộ không gian còn lại */}
    <ScrollView
      style={lcStyles.content}
      contentContainerStyle={{ padding: 16 }}
    >
      {Array.from({ length: 20 }).map((_, i) => (
        <Text key={i} style={{ marginBottom: 8 }}>
          Content line {i + 1}
        </Text>
      ))}
    </ScrollView>

    {/* Footer — kích thước cố định */}
    <View style={lcStyles.footer}>
      <Text style={lcStyles.footerText}>Footer</Text>
    </View>
  </View>
);

const lcStyles = StyleSheet.create({
  header: {
    height: 60,
    backgroundColor: "#007AFF",
    justifyContent: "center",
    alignItems: "center",
  },
  headerText: { color: "#fff", fontSize: 18, fontWeight: "bold" },
  content: { flex: 1 },
  footer: {
    height: 50,
    backgroundColor: "#333",
    justifyContent: "center",
    alignItems: "center",
  },
  footerText: { color: "#fff" },
});
```

### 3.3 Grid Layout

```tsx
import React from "react";
import { View, Text, StyleSheet, useWindowDimensions } from "react-native";

const GRID_GAP = 8;

const GridLayout = () => {
  const { width } = useWindowDimensions();
  const numColumns = width > 768 ? 4 : 2;
  const itemWidth = (width - 32 - GRID_GAP * (numColumns - 1)) / numColumns;

  return (
    <View style={gridStyles.container}>
      {Array.from({ length: 8 }).map((_, i) => (
        <View key={i} style={[gridStyles.item, { width: itemWidth }]}>
          <Text style={gridStyles.itemText}>{i + 1}</Text>
        </View>
      ))}
    </View>
  );
};

const gridStyles = StyleSheet.create({
  container: {
    flexDirection: "row",
    flexWrap: "wrap",
    padding: 16,
    gap: GRID_GAP,
  },
  item: {
    height: 100,
    backgroundColor: "#007AFF",
    borderRadius: 12,
    justifyContent: "center",
    alignItems: "center",
  },
  itemText: {
    color: "#fff",
    fontSize: 20,
    fontWeight: "bold",
  },
});
```

## 4. Flexbox Cheat Sheet

```
┌─── Container (Parent) ──────────────────────────┐
│                                                   │
│  flexDirection   → column | row | *-reverse       │
│  justifyContent  → Main Axis alignment            │
│  alignItems      → Cross Axis alignment           │
│  flexWrap        → nowrap | wrap                   │
│  alignContent    → Multi-line cross alignment      │
│  gap / rowGap / columnGap                          │
│                                                   │
├─── Item (Child) ─────────────────────────────────┤
│                                                   │
│  flex            → Grow ratio (single number)     │
│  flexGrow        → Ability to grow (0 = no)       │
│  flexShrink      → Ability to shrink (0 = no)     │
│  flexBasis       → Initial size before flex       │
│  alignSelf       → Override alignItems            │
│                                                   │
├─── Position ─────────────────────────────────────┤
│                                                   │
│  position: 'relative' (default)                   │
│  position: 'absolute' → top/right/bottom/left     │
│                                                   │
└───────────────────────────────────────────────────┘
```

```tsx
// exercises/Dashboard.tsx
import React from "react";
import {
  View,
  Text,
  StyleSheet,
  useWindowDimensions,
  ScrollView,
} from "react-native";

interface StatCard {
  title: string;
  value: string;
  color: string;
}

const stats: StatCard[] = [
  { title: "Users", value: "12,345", color: "#007AFF" },
  { title: "Revenue", value: "$45,678", color: "#28a745" },
  { title: "Orders", value: "890", color: "#f9ca24" },
  { title: "Growth", value: "+15%", color: "#6c5ce7" },
];

const Dashboard = () => {
  const { width } = useWindowDimensions();
  const cols = width > 768 ? 4 : 2;
  const cardWidth = (width - 32 - 8 * (cols - 1)) / cols;

  return (
    <ScrollView style={{ flex: 1, backgroundColor: "#f5f5f5" }}>
      {/* Stats Grid */}
      <View style={dbStyles.grid}>
        {stats.map((stat) => (
          <View
            key={stat.title}
            style={[dbStyles.statCard, { width: cardWidth }]}
          >
            <View style={[dbStyles.statDot, { backgroundColor: stat.color }]} />
            <Text style={dbStyles.statTitle}>{stat.title}</Text>
            <Text style={dbStyles.statValue}>{stat.value}</Text>
          </View>
        ))}
      </View>

      {/* Two column layout */}
      <View
        style={[dbStyles.twoCol, width < 768 && { flexDirection: "column" }]}
      >
        <View style={[dbStyles.panel, width >= 768 && { flex: 2 }]}>
          <Text style={dbStyles.panelTitle}>Recent Activity</Text>
        </View>
        <View style={[dbStyles.panel, width >= 768 && { flex: 1 }]}>
          <Text style={dbStyles.panelTitle}>Notifications</Text>
        </View>
      </View>
    </ScrollView>
  );
};

const dbStyles = StyleSheet.create({
  grid: { flexDirection: "row", flexWrap: "wrap", padding: 16, gap: 8 },
  statCard: {
    backgroundColor: "#fff",
    borderRadius: 12,
    padding: 16,
    height: 100,
    justifyContent: "space-between",
  },
  statDot: { width: 8, height: 8, borderRadius: 4 },
  statTitle: { fontSize: 13, color: "#888" },
  statValue: { fontSize: 24, fontWeight: "bold", color: "#333" },
  twoCol: {
    flexDirection: "row",
    paddingHorizontal: 16,
    gap: 8,
    marginBottom: 16,
  },
  panel: {
    backgroundColor: "#fff",
    borderRadius: 12,
    padding: 16,
    minHeight: 200,
  },
  panelTitle: { fontSize: 16, fontWeight: "600" },
});

export default Dashboard;
```
